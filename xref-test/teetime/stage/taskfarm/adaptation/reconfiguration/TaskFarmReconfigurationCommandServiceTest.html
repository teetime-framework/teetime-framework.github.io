16  package teetime.stage.taskfarm.adaptation.reconfiguration;
17  
18  import static org.hamcrest.Matchers.*;
19  import static org.junit.Assert.*;
20  
21  import org.junit.Test;
22  
23  import teetime.stage.basic.AbstractFilter;
24  import teetime.stage.taskfarm.DynamicTaskFarmStage;
25  import teetime.stage.taskfarm.ITaskFarmDuplicable;
26  
27  public class TaskFarmReconfigurationCommandServiceTest {
28  
29  	@SuppressWarnings({ "unchecked", "rawtypes" })
30  	@Test
31  	public void executionPlanTestA() {
32  		DynamicTaskFarmStage taskFarmStage = createTaskFarm();
33  		TaskFarmReconfigurationCommandService commandService = new TaskFarmReconfigurationCommandService(taskFarmStage);
34  
35  		assertThat(commandService.decideExecutionPlan(0.5d), is(equalTo(TaskFarmReconfigurationCommand.ADD)));
36  		for (int i = 0; i < taskFarmStage.getConfiguration().getMaxSamplesUntilRemove(); i++) {
37  			assertThat(commandService.decideExecutionPlan(0.1d), is(equalTo(TaskFarmReconfigurationCommand.NONE)));
38  		}
39  		assertThat(commandService.decideExecutionPlan(0.1d), is(equalTo(TaskFarmReconfigurationCommand.REMOVE)));
40  	}
41  
42  	@SuppressWarnings({ "unchecked", "rawtypes" })
43  	@Test
44  	public void executionPlanTestB() {
45  		DynamicTaskFarmStage taskFarmStage = createTaskFarm();
46  		TaskFarmReconfigurationCommandService commandService = new TaskFarmReconfigurationCommandService(taskFarmStage);
47  
48  		assertThat(commandService.decideExecutionPlan(0.5d), is(equalTo(TaskFarmReconfigurationCommand.ADD)));
49  		assertThat(commandService.decideExecutionPlan(0.1d), is(equalTo(TaskFarmReconfigurationCommand.NONE)));
50  		assertThat(commandService.decideExecutionPlan(0.1d), is(equalTo(TaskFarmReconfigurationCommand.NONE)));
51  		assertThat(commandService.decideExecutionPlan(0.3d), is(equalTo(TaskFarmReconfigurationCommand.ADD)));
52  		assertThat(commandService.decideExecutionPlan(0.1d), is(equalTo(TaskFarmReconfigurationCommand.NONE)));
53  		assertThat(commandService.decideExecutionPlan(0.1d), is(equalTo(TaskFarmReconfigurationCommand.NONE)));
54  		assertThat(commandService.decideExecutionPlan(0.1d), is(equalTo(TaskFarmReconfigurationCommand.NONE)));
55  		assertThat(commandService.decideExecutionPlan(0.3d), is(equalTo(TaskFarmReconfigurationCommand.ADD)));
56  		for (int i = 0; i < taskFarmStage.getConfiguration().getMaxSamplesUntilRemove(); i++) {
57  			assertThat(commandService.decideExecutionPlan(0.1d), is(equalTo(TaskFarmReconfigurationCommand.NONE)));
58  		}
59  		assertThat(commandService.decideExecutionPlan(0.1d), is(equalTo(TaskFarmReconfigurationCommand.REMOVE)));
60  	}
61  
62  	@SuppressWarnings({ "rawtypes", "unchecked" })
63  	private DynamicTaskFarmStage createTaskFarm() {
64  		DynamicTaskFarmStage taskFarmStage = new DynamicTaskFarmStage(new DummyDuplicableStage(), 1);
65  		return taskFarmStage;
66  	}
67  
68  	@SuppressWarnings("rawtypes")
69  	private class DummyDuplicableStage extends AbstractFilter implements ITaskFarmDuplicable {
70  		@Override
71  		public ITaskFarmDuplicable duplicate() {
72  			return null;
73  		}
74  
75  		@Override
76  		protected void execute(final Object element) {
77  			// nothing to do
78  		}
79  	}
80  }
