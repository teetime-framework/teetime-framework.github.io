16  package teetime.stage.taskfarm.monitoring.extraction;
17  
18  import static org.hamcrest.Matchers.equalTo;
19  import static org.hamcrest.Matchers.is;
20  import static org.hamcrest.Matchers.stringContainsInOrder;
21  import static org.junit.Assert.assertThat;
22  
23  import java.util.LinkedList;
24  import java.util.List;
25  
26  import org.junit.Test;
27  
28  import teetime.stage.taskfarm.monitoring.PipeMonitoringService;
29  
30  public class StackedTimePullThroughput2DTest {
31  
32  	@Test
33  	public void testWorkingService() {
34  		PipeMonitoringService service = ExtractorTestHelper.generate4PipeMonitoringServiceWithBehavior();
35  
36  		AbstractMonitoringDataExporter extraction = new StackedTimePullThroughput2D(service, null);
37  		String result = extraction.extractToString();
38  
39  		List<String> outputValues = new LinkedList<String>();
40  		outputValues.add("time,pullthroughput0,pullthroughput1,pullthroughput2,pullthroughput3");
41  		// as the exact processing time is slightly nondeterministic, we only check for the other values
42  		outputValues.add(",6,0,0,0");
43  		outputValues.add(",11,0,0,0");
44  		outputValues.add(",16,6,0,0");
45  		outputValues.add(",21,0,6,0");
46  		outputValues.add(",26,0,11,0");
47  		outputValues.add(",31,0,0,0");
48  		outputValues.add(",36,0,0,6");
49  		assertThat(result, stringContainsInOrder(outputValues));
50  	}
51  
52  	@Test
53  	public void testEmptyService() {
54  		PipeMonitoringService service = ExtractorTestHelper.generateEmpty5PipeMonitoringService();
55  
56  		AbstractMonitoringDataExporter extraction = new StackedTimePullThroughput2D(service, null);
57  		String result = extraction.extractToString();
58  
59  		String header = "time,pullthroughput0,pullthroughput1,pullthroughput2,pullthroughput3,pullthroughput4"
60  				+ System.getProperty("line.separator");
61  		assertThat(result, is(equalTo(header)));
62  	}
63  }
