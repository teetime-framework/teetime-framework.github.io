16  package teetime.stage;
17  
18  import static org.hamcrest.collection.IsCollectionWithSize.hasSize;
19  import static org.hamcrest.collection.IsEmptyCollection.empty;
20  import static org.hamcrest.collection.IsIterableContainingInOrder.contains;
21  import static org.hamcrest.core.Is.is;
22  import static org.junit.Assert.assertThat;
23  import static teetime.framework.test.StageTester.test;
24  
25  import java.util.ArrayList;
26  import java.util.List;
27  
28  import org.junit.Before;
29  import org.junit.Test;
30  
31  import teetime.framework.Configuration;
32  import teetime.framework.Execution;
33  
34  /**
35   * @author Nils Christian Ehmke
36   */
37  public class InstanceOfFilterTest {
38  
39  	private InstanceOfFilter<Object, Clazz> filter;
40  
41  	@Before
42  	public void initializeFilter() {
43  		filter = new InstanceOfFilter<Object, Clazz>(Clazz.class);
44  	}
45  
46  	@Test
47  	public void filterShouldForwardCorrectTypes() {
48  		final List<Clazz> results = new ArrayList<InstanceOfFilterTest.Clazz>();
49  		final Object clazz = new Clazz();
50  
51  		test(filter)
52  				.and().send(clazz).to(filter.getInputPort())
53  				.and().receive(results).from(filter.getMatchedOutputPort())
54  				.start();
55  
56  		assertThat(results, contains(clazz));
57  	}
58  
59  	@Test
60  	public void filterShouldForwardSubTypes() {
61  		final List<Clazz> results = new ArrayList<InstanceOfFilterTest.Clazz>();
62  		final Object clazz = new SubClazz();
63  
64  		test(filter)
65  				.and().send(clazz).to(filter.getInputPort())
66  				.and().receive(results).from(filter.getMatchedOutputPort())
67  				.start();
68  
69  		assertThat(results, contains(clazz));
70  	}
71  
72  	@Test
73  	public void filterShouldDropInvalidTypes() {
74  		final List<Clazz> results = new ArrayList<InstanceOfFilterTest.Clazz>();
75  		final Object object = new Object();
76  
77  		test(filter)
78  				.and().send(object).to(filter.getInputPort())
79  				.and().receive(results).from(filter.getMatchedOutputPort())
80  				.start();
81  
82  		assertThat(results, is(empty()));
83  	}
84  
85  	@Test
86  	public void filterShouldWorkWithMultipleInput() {
87  		final List<Clazz> results = new ArrayList<InstanceOfFilterTest.Clazz>();
88  		final List<Object> inputObjects = new ArrayList<Object>();
89  
90  		inputObjects.add(new Object());
91  		inputObjects.add(new Clazz());
92  		inputObjects.add(new Object());
93  		inputObjects.add(new SubClazz());
94  		inputObjects.add(new Object());
95  
96  		test(filter)
97  				.and().send(inputObjects).to(filter.getInputPort())
98  				.and().receive(results).from(filter.getMatchedOutputPort())
99  				.start();
100 
101 		assertThat(results, hasSize(2));
102 	}
103 
104 	private static class Clazz {
105 	}
106 
107 	private static class SubClazz extends Clazz {
108 	}
109 
110 	@Test
111 	public void filterShouldSendToBothOutputPorts() {
112 		InstanceOfFilterTestConfig config = new InstanceOfFilterTestConfig();
113 		Execution<InstanceOfFilterTestConfig> execution = new Execution<InstanceOfFilterTestConfig>(config);
114 		execution.executeBlocking();
115 	}
116 
117 	private static class InstanceOfFilterTestConfig extends Configuration {
118 
119 		public InstanceOfFilterTestConfig() {
120 			InitialElementProducer<Object> elementProducer = new InitialElementProducer<Object>();
121 			InstanceOfFilter<Object, Clazz> instanceOfFilter = new InstanceOfFilter<Object, Clazz>(Clazz.class);
122 			CollectorSink<Clazz> clazzCollector = new CollectorSink<Clazz>();
123 			CollectorSink<Object> mismatchedCollector = new CollectorSink<Object>();
124 
125 			connectPorts(elementProducer.getOutputPort(), instanceOfFilter.getInputPort());
126 			connectPorts(instanceOfFilter.getMatchedOutputPort(), clazzCollector.getInputPort());
127 			connectPorts(instanceOfFilter.getMismatchedOutputPort(), mismatchedCollector.getInputPort());
128 
129 		}
130 	}
131 
132 }
