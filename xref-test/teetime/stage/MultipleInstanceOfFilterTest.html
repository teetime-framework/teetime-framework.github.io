16  package teetime.stage;
17  
18  import static org.hamcrest.Matchers.contains;
19  import static org.hamcrest.Matchers.empty;
20  import static org.hamcrest.Matchers.is;
21  import static org.hamcrest.Matchers.not;
22  import static org.junit.Assert.assertThat;
23  
24  import java.util.ArrayList;
25  import java.util.Arrays;
26  import java.util.List;
27  
28  import org.junit.Test;
29  
30  import teetime.framework.OutputPort;
31  import teetime.framework.test.StageTester;
32  
33  /**
34   * @author Nils Christian Ehmke
35   */
36  public class MultipleInstanceOfFilterTest {
37  
38  	@Test
39  	@SuppressWarnings("unchecked")
40  	public void filteringForSingleTypeShouldWork() {
41  		final MultipleInstanceOfFilter<Object> filter = new MultipleInstanceOfFilter<Object>();
42  		final List<Object> inputObjects = new ArrayList<Object>(Arrays.asList("1", 1.5f, "2", 2.5f, "3", 3.5f));
43  		final List<String> receivedStrings = new ArrayList<String>();
44  
45  		StageTester.test(filter).and().send(inputObjects).to(filter.getInputPort()).and().receive(receivedStrings).from(filter.getOutputPortForType(String.class)).start();
46  
47  		assertThat(receivedStrings, is(not(empty())));
48  		assertThat(receivedStrings, contains("1", "2", "3"));
49  	}
50  
51  	@Test
52  	@SuppressWarnings("unchecked")
53  	public void filteringForMultipleTypesShouldWork() {
54  		final MultipleInstanceOfFilter<Number> filter = new MultipleInstanceOfFilter<Number>();
55  		final List<Number> inputObjects = new ArrayList<Number>(Arrays.asList(1, 1.5f, 2, 2.5f, 3, 3.5f));
56  		final List<Integer> integers = new ArrayList<Integer>();
57  		final List<Float> floats = new ArrayList<Float>();
58  
59  		StageTester.test(filter).and().send(inputObjects).to(filter.getInputPort()).and().receive(integers).from(filter.getOutputPortForType(Integer.class)).and()
60  				.receive(floats).from(filter.getOutputPortForType(Float.class)).start();
61  
62  		assertThat(integers, contains(1, 2, 3));
63  		assertThat(floats, contains(1.5f, 2.5f, 3.5f));
64  	}
65  
66  	@Test
67  	public void outputPortForSameTypeShouldBeCached() {
68  		final MultipleInstanceOfFilter<Number> filter = new MultipleInstanceOfFilter<Number>();
69  
70  		final OutputPort<Float> firstPort = filter.getOutputPortForType(Float.class);
71  		final OutputPort<Float> secondPort = filter.getOutputPortForType(Float.class);
72  
73  		assertThat(firstPort, is(secondPort));
74  	}
75  
76  }
