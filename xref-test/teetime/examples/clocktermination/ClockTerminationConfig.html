1   package teetime.examples.clocktermination;
2   
3   import teetime.framework.Configuration;
4   import teetime.stage.Clock;
5   import teetime.stage.InitialElementProducer;
6   import teetime.stage.basic.Sink;
7   
8   /**
9    * This configuration is special because it did not terminate in previous versions of teetime.
10   * In these versions, all stages were collected which were reachable from the stage which was connected first by <code>connectPorts()</code>.
11   * Since the <code>firstProducer</code> cannot be reached by this stage (here: the clock stage),
12   * the <code>firstProducer</code> is not recognized as a stage and especially not as a producer stage.
13   * Since the clock is an infinite producer, the execution waits for its termination an infinite amount of time.
14   *
15   * @author Christian Wulf
16   *
17   */
18  public class ClockTerminationConfig extends Configuration {
19  
20  	public ClockTerminationConfig() {
21  		InitialElementProducer<Integer> firstProducer = new InitialElementProducer<Integer>(0, 1);
22  		Clock clock = new Clock();
23  
24  		Sink<Integer> firstSink = new Sink<Integer>();
25  		Sink<Long> secondSink = new Sink<Long>();
26  
27  		connectPorts(firstProducer.getOutputPort(), firstSink.getInputPort());
28  		connectPorts(clock.getOutputPort(), secondSink.getInputPort());
29  	}
30  }
