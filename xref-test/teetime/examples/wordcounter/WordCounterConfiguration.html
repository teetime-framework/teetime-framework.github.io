16  package teetime.examples.wordcounter;
17  
18  import java.io.File;
19  import java.util.ArrayList;
20  import java.util.List;
21  
22  import teetime.framework.AbstractPort;
23  import teetime.framework.Configuration;
24  import teetime.framework.MonitoringThread;
25  import teetime.stage.CountingMapMerger;
26  import teetime.stage.InitialElementProducer;
27  import teetime.stage.basic.distributor.Distributor;
28  import teetime.stage.basic.distributor.strategy.NonBlockingRoundRobinStrategy;
29  import teetime.stage.basic.merger.Merger;
30  import teetime.stage.io.File2SeqOfWords;
31  import teetime.stage.string.WordCounter;
32  import teetime.stage.util.CountingMap;
33  
34  /**
35   * A simple configuration, which counts the words of a set of files.
36   * The execution of this configuration is demonstrated in {@link WordCountingTest}.
37   *
38   * This configuration is divided into three parts. The first part reads files and distributes them to different {@link WordCounter} instances.
39   * The second part are a certain number of WordCounter instances. On construction of this class the number of concurrent WordCounter instances is specified with the
40   * threads parameter.
41   * The third and last part collects the results from all WordCounter instances and merges them. This final result can be read afterwards.
42   *
43   *
44   * @author Nelson Tavares de Sousa
45   *
46   */
47  public class WordCounterConfiguration extends Configuration {
48  
49  	// Last stage is saved as field, to retrieve the result after execution.
50  	private final CountingMapMerger<String> result = new CountingMapMerger<String>();
51  
52  	private final List<AbstractPort<?>> distributorPorts = new ArrayList<AbstractPort<?>>();
53  	private final List<AbstractPort<?>> mergerPorts = new ArrayList<AbstractPort<?>>();
54  
55  	private final MonitoringThread monitoringThread;
56  
57  	private final Distributor<String> distributor;
58  
59  	public WordCounterConfiguration(final int threads, final File... input) {
60  		// First part of the config
61  		final InitialElementProducer<File> init = new InitialElementProducer<File>(input);
62  		// final File2Lines f2b = new File2Lines();
63  		final File2SeqOfWords f2b = new File2SeqOfWords("UTF-8", 512);
64  		distributor = new Distributor<String>(new NonBlockingRoundRobinStrategy());
65  
66  		// last part
67  		final Merger<CountingMap<String>> merger = new Merger<CountingMap<String>>();
68  		// CountingMapMerger (already as field)
69  
70  		// Connecting the stages of the first part of the config
71  		connectPorts(init.getOutputPort(), f2b.getInputPort());
72  		connectPorts(f2b.getOutputPort(), distributor.getInputPort());
73  
74  		monitoringThread = new MonitoringThread();
75  
76  		// Middle part... multiple instances of WordCounter are created and connected to the merger and distrubuter stages
77  		for (int i = 0; i < threads; i++) {
78  			// final InputPortSizePrinter<String> inputPortSizePrinter = new InputPortSizePrinter<String>();
79  			final WordCounter wc = new WordCounter();
80  			// intraFact.create(inputPortSizePrinter.getOutputPort(), wc.getInputPort());
81  			final WordCounter threadableStage = wc;
82  
83  			connectPorts(distributor.getNewOutputPort(), threadableStage.getInputPort(), 1000);
84  			connectPorts(wc.getOutputPort(), merger.getNewInputPort());
85  			// Add WordCounter as a threadable stage, so it runs in its own thread
86  			threadableStage.getInputPort().getOwningStage().declareActive();
87  
88  			distributorPorts.add(threadableStage.getInputPort());
89  			mergerPorts.add(wc.getOutputPort());
90  
91  			monitoringThread.addPort(threadableStage.getInputPort());
92  		}
93  
94  		// Connect the stages of the last part
95  		connectPorts(merger.getOutputPort(), result.getInputPort());
96  
97  		// Add the first and last part to the threadable stages
98  		init.declareActive();
99  		merger.declareActive();
100 	}
101 
102 	public MonitoringThread getMonitoringThread() {
103 		return monitoringThread;
104 	}
105 
106 	// Further methods are allowed. For e.g. it is possible to read data from certain stages.
107 	public CountingMap<String> getResult() {
108 		return result.getResult();
109 	}
110 
111 	public List<AbstractPort<?>> getDistributorPorts() {
112 		return distributorPorts;
113 	}
114 
115 	public List<AbstractPort<?>> getMergerPorts() {
116 		return mergerPorts;
117 	}
118 
119 	public Distributor<String> getDistributor() {
120 		return distributor;
121 	}
122 
123 }
