16  package teetime.examples.listwordcounter;
17  
18  import java.io.IOException;
19  import java.util.Arrays;
20  import java.util.List;
21  
22  import org.junit.Assert;
23  import org.junit.Test;
24  
25  import teetime.framework.*;
26  import teetime.stage.CollectorSink;
27  import teetime.stage.InitialElementProducer;
28  import teetime.stage.string.*;
29  import teetime.stage.util.CountingMap;
30  
31  /**
32   * Execute stages that count count the occurrences of words in a list of strings.
33   * Procedure: read string list > to lower case > remove punctuation > count word occurrences
34   *
35   * @author Sören Henning
36   *
37   */
38  public class ListWordCounterTest {
39  
40  	private static final List<String> STRINGS = Arrays.asList("Hello World!", "Hello TeeTime.", "Bye world");
41  
42  	public ListWordCounterTest() {}
43  
44  	@Test
45  	public void executeTestWithDefaultConfiguration() throws IOException {
46  		final ListWordCounterConfiguration configuration = new ListWordCounterConfiguration(STRINGS);
47  		final Execution<ListWordCounterConfiguration> execution = new Execution<>(configuration);
48  		execution.executeBlocking();
49  
50  		CountingMap<String> map = configuration.getCountingMap().orElse(new CountingMap<>());
51  
52  		Assert.assertEquals(2, map.get("hello"));
53  		Assert.assertEquals(2, map.get("world"));
54  		Assert.assertEquals(1, map.get("teetime"));
55  		Assert.assertEquals(1, map.get("bye"));
56  	}
57  
58  	@Test
59  	public void executeTestWithBuilderBasedConfiguration() throws IOException {
60  		final ListWordCounterConfigurationFromBuilder configuration = new ListWordCounterConfigurationFromBuilder(STRINGS);
61  		final Execution<ListWordCounterConfigurationFromBuilder> execution = new Execution<>(configuration);
62  		execution.executeBlocking();
63  
64  		CountingMap<String> map = configuration.getCountingMap().orElse(new CountingMap<>());
65  
66  		Assert.assertEquals(2, map.get("hello"));
67  		Assert.assertEquals(2, map.get("world"));
68  		Assert.assertEquals(1, map.get("teetime"));
69  		Assert.assertEquals(1, map.get("bye"));
70  	}
71  
72  	@Test
73  	public void executeTestWithConfigurationCreatedByBuilder() throws IOException {
74  		final CollectorSink<CountingMap<String>> collector = new CollectorSink<>();
75  
76  		final Configuration configuration = ConfigurationBuilder
77  				.from(new InitialElementProducer<>(STRINGS))
78  				.to(new ToLowerCase())
79  				.to(new WordcharacterFilter())
80  				.to(new WordCounter())
81  				.end(collector);
82  		final Execution<Configuration> execution = new Execution<Configuration>(configuration);
83  		execution.executeBlocking();
84  
85  		final List<CountingMap<String>> maps = collector.getElements();
86  
87  		Assert.assertTrue(maps.size() > 0);
88  
89  		CountingMap<String> map = maps.get(0);
90  
91  		Assert.assertEquals(2, map.get("hello"));
92  		Assert.assertEquals(2, map.get("world"));
93  		Assert.assertEquals(1, map.get("teetime"));
94  		Assert.assertEquals(1, map.get("bye"));
95  	}
96  
97  }
