16  package teetime.examples.tokenizer;
17  
18  import java.io.File;
19  
20  import teetime.framework.Configuration;
21  import teetime.stage.ByteArray2String;
22  import teetime.stage.CipherStage;
23  import teetime.stage.CipherStage.CipherMode;
24  import teetime.stage.Counter;
25  import teetime.stage.InitialElementProducer;
26  import teetime.stage.ZipByteArray;
27  import teetime.stage.ZipByteArray.ZipMode;
28  import teetime.stage.io.File2ByteArray;
29  import teetime.stage.string.Tokenizer;
30  
31  public class TokenizerConfiguration extends Configuration {
32  
33  	private final Counter<String> counter;
34  
35  	public TokenizerConfiguration(final String inputFile, final String password) {
36  		final File input = new File(inputFile);
37  
38  		final InitialElementProducer<File> init = new InitialElementProducer<File>(input);
39  		final File2ByteArray f2b = new File2ByteArray();
40  		final ZipByteArray decomp = new ZipByteArray(ZipMode.DECOMP);
41  		final CipherStage decrypt = new CipherStage(password, CipherMode.DECRYPT);
42  		final ByteArray2String b2s = new ByteArray2String();
43  		final Tokenizer tokenizer = new Tokenizer(" ");
44  		this.counter = new Counter<String>();
45  
46  		connectPorts(init.getOutputPort(), f2b.getInputPort());
47  		connectPorts(f2b.getOutputPort(), decomp.getInputPort());
48  		connectPorts(decomp.getOutputPort(), decrypt.getInputPort());
49  		connectPorts(decrypt.getOutputPort(), b2s.getInputPort());
50  		connectPorts(b2s.getOutputPort(), tokenizer.getInputPort());
51  		connectPorts(tokenizer.getOutputPort(), this.counter.getInputPort());
52  
53  	}
54  
55  	public int getTokenCount() {
56  		return this.counter.getNumElementsPassed();
57  	}
58  
59  }
