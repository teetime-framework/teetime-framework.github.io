16  package teetime.framework;
17  
18  import static org.hamcrest.MatcherAssert.*;
19  import static org.hamcrest.Matchers.*;
20  
21  import java.util.Arrays;
22  import java.util.List;
23  
24  import org.junit.Test;
25  
26  import teetime.framework.pipe.BoundedSynchedPipe;
27  import teetime.framework.pipe.BoundedSynchedPipeFactory;
28  import teetime.stage.CollectorSink;
29  import teetime.stage.InitialElementProducer;
30  
31  public class ConfigurationTest {
32  
33  	public static final int CUSTOM_PIPE_CAPACITY = 1024;
34  	public static final Integer[] INPUT_ELEMENTS = { 1, 2, 3 }; // NOPMD
35  	public static final List<Integer> EXPECTED_OUTPUT_ELEMENTS = Arrays.asList(INPUT_ELEMENTS); // NOPMD
36  
37  	private static class ConfigurationWithProvidedOrCustomPipe extends Configuration {
38  
39  		private final CollectorSink<Integer> collectorSink;
40  
41  		public ConfigurationWithProvidedOrCustomPipe(final boolean withProvidedPipe) {
42  			InitialElementProducer<Integer> producer = new InitialElementProducer<Integer>(INPUT_ELEMENTS);
43  			CollectorSink<Integer> collectorSink = new CollectorSink<Integer>();
44  
45  			collectorSink.declareActive();
46  
47  			if (withProvidedPipe) {
48  				connectPorts(producer.getOutputPort(), collectorSink.getInputPort());
49  			} else {
50  				// AbstractPipe<Integer> pipe = new BoundedSynchedPipe<Integer>(producer.getOutputPort(), collectorSink.getInputPort(), 512);
51  				// registerCustomPipe(pipe);
52  				connectPorts(producer.getOutputPort(), collectorSink.getInputPort(), CUSTOM_PIPE_CAPACITY, BoundedSynchedPipeFactory.INSTANCE);
53  			}
54  
55  			this.collectorSink = collectorSink;
56  		}
57  
58  		List<Integer> getOutputElements() {
59  			return collectorSink.getElements();
60  		}
61  	}
62  
63  	@Test
64  	public void configWithProvidedPipeShouldBeExecutable() throws Exception {
65  		ConfigurationWithProvidedOrCustomPipe configuration = new ConfigurationWithProvidedOrCustomPipe(true);
66  		Execution<Configuration> execution = new Execution<Configuration>(configuration);
67  		execution.executeBlocking();
68  		assertThat(configuration.getOutputElements(), is(EXPECTED_OUTPUT_ELEMENTS));
69  		assertThat(configuration.collectorSink.inputPort.pipe, is(instanceOf(BoundedSynchedPipe.class)));
70  		assertThat(configuration.collectorSink.inputPort.pipe.capacity(), is(CompositeStage.DEFAULT_PIPE_CAPACITY));
71  	}
72  
73  	@Test
74  	public void configWithCustomPipeShouldBeExecutable() throws Exception {
75  		ConfigurationWithProvidedOrCustomPipe configuration = new ConfigurationWithProvidedOrCustomPipe(false);
76  		Execution<Configuration> execution = new Execution<Configuration>(configuration);
77  		execution.executeBlocking();
78  		assertThat(configuration.getOutputElements(), is(EXPECTED_OUTPUT_ELEMENTS));
79  		assertThat(configuration.collectorSink.inputPort.pipe, is(instanceOf(BoundedSynchedPipe.class)));
80  		assertThat(configuration.collectorSink.inputPort.pipe.capacity(), is(CUSTOM_PIPE_CAPACITY));
81  	}
82  }
