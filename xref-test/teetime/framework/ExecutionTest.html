43  	@Test
44  	public void testExecuteNonBlocking() throws Exception {
45  		StopWatch watch = new StopWatch();
46  		watch.start();
47  		execution.executeNonBlocking();
48  		watch.end();
49  
50  		assertThat(watch.getDurationInMs(), is(lessThan(DELAY_IN_MS)));
51  		assertFalse(execution.getConfiguration().delay.finished);
52  
53  		execution.waitForTermination();
54  		assertTrue(execution.getConfiguration().delay.finished);
55  	}

57  	@Test
58  	public void testExecuteBlocking() {
59  		StopWatch watch = new StopWatch();
60  		watch.start();
61  		execution.executeBlocking();
62  		watch.end();
63  
64  		assertThat(watch.getDurationInMs(), is(greaterThanOrEqualTo(DELAY_IN_MS - ABSOLUTE_MAX_ERROR_IN_MS)));
65  	}

67  	private static class TestConfig extends Configuration {
68  		public final DelayAndTerminate delay;
69  
70  		public TestConfig() {
71  			final InitialElementProducer<String> init = new InitialElementProducer<String>("Hello");
72  			delay = new DelayAndTerminate(DELAY_IN_MS);
73  			connectPorts(init.getOutputPort(), delay.getInputPort());
74  		}
75  	}
76  
77  	private static class DelayAndTerminate extends AbstractConsumerStage<String> {
78  
79  		private final long delayInMs;
80  
81  		public boolean finished;
82  
83  		public DelayAndTerminate(final long delayInMs) {
84  			super();
85  			this.delayInMs = delayInMs;
86  		}
87  
88  		@Override
89  		protected void execute(final String element) throws InterruptedException {
90  			Thread.sleep(delayInMs);
91  			finished = true;
92  		}
93  
94  	}

97  	public void testInstantiatePipes() throws Exception {
98  		Execution<AnalysisTestConfig> interAnalysis = new Execution<AnalysisTestConfig>(new AnalysisTestConfig(true));
99  		assertThat(interAnalysis.getConfiguration().init.getOwningThread(), is(not(interAnalysis.getConfiguration().sink.getOwningThread())));
100 
101 		Execution<AnalysisTestConfig> intraAnalysis = new Execution<AnalysisTestConfig>(new AnalysisTestConfig(false));
102 		assertThat(intraAnalysis.getConfiguration().init.getOwningThread(), is(intraAnalysis.getConfiguration().sink.getOwningThread()));
103 	}
