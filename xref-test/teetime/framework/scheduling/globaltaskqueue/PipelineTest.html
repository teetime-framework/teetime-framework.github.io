16  package teetime.framework.scheduling.globaltaskqueue;
17  
18  import static org.hamcrest.Matchers.*;
19  import static org.junit.Assert.*;
20  
21  import java.util.Arrays;
22  import java.util.List;
23  
24  import org.junit.Test;
25  
26  import teetime.framework.Configuration;
27  import teetime.framework.ConfigurationContext;
28  import teetime.framework.Execution;
29  import teetime.framework.exceptionHandling.TerminatingExceptionListenerFactory;
30  import teetime.stage.CollectorSink;
31  import teetime.stage.Counter;
32  import teetime.stage.InitialElementProducer;
33  
34  public class PipelineTest {
35  
36  	private static class GlobalTaskQueueConfig<T> extends Configuration {
37  
38  		private static final int NUM_THREADS = 4;
39  		private static final GlobalTaskQueueScheduling SCHEDULER = new GlobalTaskQueueScheduling(NUM_THREADS);
40  		private CollectorSink<T> sink;
41  
42  		public GlobalTaskQueueConfig(final T... elements) {
43  			super(new TerminatingExceptionListenerFactory(), new ConfigurationContext(SCHEDULER));
44  			SCHEDULER.setConfiguration(this);
45  			build(elements);
46  		}
47  
48  		private void build(final T... elements) {
49  			InitialElementProducer<T> producer = new InitialElementProducer<>(elements);
50  			Counter<T> counter = new Counter<>();
51  			sink = new CollectorSink<>();
52  
53  			from(producer).to(counter).end(sink);
54  		}
55  
56  		public CollectorSink<T> getSink() {
57  			return sink;
58  		}
59  	}
60  
61  	@Test
62  	public void shouldExecutePipelineCorrectly() throws Exception {
63  		String[] inputElements = { "a", "b", "c" };
64  		GlobalTaskQueueConfig<String> config = new GlobalTaskQueueConfig<>(inputElements);
65  		Execution<GlobalTaskQueueConfig<String>> execution = new Execution<>(config);
66  		execution.executeBlocking();
67  
68  		List<String> processedElements = config.getSink().getElements();
69  		List<String> expectedElements = Arrays.asList("a", "b", "c");
70  		assertThat(processedElements, is(equalTo(expectedElements)));
71  	}
72  }
