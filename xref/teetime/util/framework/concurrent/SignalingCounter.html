16  package teetime.util.framework.concurrent;
17  
18  import com.carrotsearch.hppc.IntObjectHashMap;
19  import com.carrotsearch.hppc.IntObjectMap;
20  
21  public class SignalingCounter {
22  
23  	private final IntObjectMap<Object> conditions = new IntObjectHashMap<Object>();
24  	private int counter;
25  
26  	// synchronized methods synchronize the map and the counter
27  	// synchronized(cond) synchronizes the individual numbers for which are being waited for
28  
29  	public synchronized void inc() {
30  		counter++;
31  		conditionalNotifyAll(counter);
32  	}
33  
34  	public synchronized void inc(final SignalingCounter otherCounter) {
35  		counter += otherCounter.counter;
36  		conditionalNotifyAll(counter);
37  	}
38  
39  	public synchronized void dec() {
40  		counter--;
41  		conditionalNotifyAll(counter);
42  	}
43  
44  	private synchronized void conditionalNotifyAll(final int number) {
45  		if (conditions.containsKey(number)) {
46  			Object cond = conditions.get(number);
47  			synchronized (cond) {
48  				cond.notifyAll(); // If you came here because of FindBugs, go on, everything's ok here. You saw nothing!
49  			}
50  		}
51  	}
52  
53  	public void waitFor(final int number) throws InterruptedException {
54  		synchronized (this) {
55  			if (!conditions.containsKey(number)) {
56  				conditions.put(number, new Object());
57  			}
58  		}
59  
60  		final Object cond = conditions.get(number);
61  
62  		synchronized (cond) { // counter must be wrapped by synchronized to get the latest value
63  			while (counter != number) {
64  				cond.wait();
65  			}
66  		}
67  	}
68  
69  	@Override
70  	public String toString() {
71  		return "counter: " + counter + ", " + super.toString();
72  	}
73  }
