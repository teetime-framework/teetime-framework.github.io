16  package teetime.stage.basic.distributor.dynamic;
17  
18  import java.util.ArrayList;
19  import java.util.List;
20  
21  import teetime.framework.InputPort;
22  import teetime.framework.OutputPort;
23  import teetime.framework.pipe.BoundedSynchedPipe;
24  import teetime.framework.signal.StartingSignal;
25  import teetime.util.framework.port.PortAction;
26  import teetime.util.stage.OneTimeCondition;
27  
28  /**
29   *
30   * @author Christian Wulf
31   *
32   * @param <T>
33   *
34   * @deprecated since 3.0. Use {@link CreatePortActionDistributor} instead.
35   */
36  @Deprecated
37  public class CreatePortAction<T> implements PortAction<DynamicDistributor<T>> {
38  
39  	private final List<PortActionListener<T>> listeners = new ArrayList<PortActionListener<T>>();
40  	private final OneTimeCondition condition = new OneTimeCondition();
41  
42  	private final InputPort<T> inputPort;
43  	private final int capacity;
44  
45  	public CreatePortAction(final InputPort<T> inputPort, final int capacity) {
46  		super();
47  		this.inputPort = inputPort;
48  		this.capacity = capacity;
49  	}
50  
51  	@Override
52  	public void execute(final DynamicDistributor<T> dynamicDistributor) {
53  		OutputPort<T> newOutputPort = dynamicDistributor.getNewOutputPort();
54  
55  		new BoundedSynchedPipe<T>(newOutputPort, inputPort, capacity);
56  
57  		newOutputPort.sendSignal(new StartingSignal());
58  
59  		onOutputPortCreated(dynamicDistributor, newOutputPort);
60  		condition.signalAll();
61  	}
62  
63  	private void onOutputPortCreated(final DynamicDistributor<T> dynamicDistributor, final OutputPort<T> newOutputPort) {
64  		for (PortActionListener<T> listener : listeners) {
65  			listener.onOutputPortCreated(dynamicDistributor, newOutputPort);
66  		}
67  	}
68  
69  	/* default */ InputPort<T> getInputPort() { // for testing purposes only
70  		return inputPort;
71  	}
72  
73  	public void addPortActionListener(final PortActionListener<T> listener) {
74  		listeners.add(listener);
75  	}
76  
77  	public void waitForCompletion() throws InterruptedException {
78  		condition.await();
79  	}
80  }
