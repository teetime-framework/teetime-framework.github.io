1   package teetime.stage;
2   
3   import teetime.framework.AbstractStage;
4   import teetime.framework.InputPort;
5   import teetime.framework.OutputPort;
6   import teetime.framework.pipe.ReflexivePipe;
7   
8   public class StatelessCounter<T> extends AbstractStage {
9   
10  	private final InputPort<T> inputPort = this.createInputPort();
11  	private final InputPort<CounterContainer> counterInputPort = this.createInputPort();
12  
13  	private final OutputPort<T> outputPort = this.createOutputPort();
14  	private final OutputPort<CounterContainer> counterOutputPort = this.createOutputPort();
15  
16  	public StatelessCounter() {
17  		new ReflexivePipe<CounterContainer>(counterOutputPort, counterInputPort);
18  		counterOutputPort.send(new CounterContainer());
19  	}
20  
21  	@Override
22  	protected void execute() {
23  		T element = inputPort.receive();
24  		if (element == null) {
25  			returnNoElement();
26  		}
27  
28  		outputPort.send(element);
29  
30  		CounterContainer counterContainer = counterInputPort.receive();
31  		// null check not necessary since counterContainer is initialized in the ctor and can never be accessed from outside
32  		counterContainer.counter = counterContainer.counter + 1;
33  		counterOutputPort.send(counterContainer);
34  	}
35  
36  	public InputPort<T> getInputPort() {
37  		return inputPort;
38  	}
39  
40  	public OutputPort<T> getOutputPort() {
41  		return outputPort;
42  	}
43  
44  	private static class CounterContainer {
45  		long counter;
46  	}
47  
48  }
