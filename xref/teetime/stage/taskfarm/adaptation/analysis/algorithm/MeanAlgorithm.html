16  package teetime.stage.taskfarm.adaptation.analysis.algorithm;
17  
18  import teetime.stage.taskfarm.TaskFarmConfiguration;
19  import teetime.stage.taskfarm.adaptation.analysis.AbstractThroughputAlgorithm;
20  import teetime.stage.taskfarm.adaptation.history.ThroughputHistory;
21  
22  /**
23   * Represents the analysis of the throughput of a certain amount of
24   * items and predicts the next value based on the mean value.
25   *
26   * @author Christian Claus Wiechmann
27   *
28   */
29  public class MeanAlgorithm extends AbstractThroughputAlgorithm {
30  
31  	/**
32  	 * Constructor.
33  	 *
34  	 * @param configuration
35  	 *            TaskFarmConfiguration of the Task Farm which
36  	 *            this algorithm is used for
37  	 */
38  	public MeanAlgorithm(final TaskFarmConfiguration<?, ?, ?> configuration) {
39  		super(configuration);
40  	}
41  
42  	@Override
43  	protected double doAnalysis(final ThroughputHistory history) {
44  		double sum = 0;
45  
46  		for (int i = 1; i <= this.window; i++) {
47  			final double current = history.getThroughputOfEntry(i);
48  			sum += current;
49  		}
50  
51  		return sum / this.window;
52  	}
53  }
