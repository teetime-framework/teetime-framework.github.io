16  package teetime.stage.taskfarm.adaptation.reconfiguration;
17  
18  /**
19   * Represents the current plan of the task farm reconfiguration service.
20   * Possible values are addition or removal of a stage. Furthermore, no
21   * action at all can be taken.
22   *
23   * @author Christian Claus Wiechmann
24   */
25  enum TaskFarmReconfigurationCommand {
26  	/** Represents the addition of a new stage. **/
27  	ADD,
28  	/** Represents the removal of an existing stage. **/
29  	REMOVE,
30  	/** Represents that no action at all should be taken. **/
31  	NONE
32  }
