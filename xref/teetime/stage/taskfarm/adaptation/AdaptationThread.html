16  package teetime.stage.taskfarm.adaptation;
17  
18  import org.slf4j.Logger;
19  import org.slf4j.LoggerFactory;
20  
21  import teetime.stage.taskfarm.*;
22  import teetime.stage.taskfarm.adaptation.analysis.TaskFarmAnalysisService;
23  import teetime.stage.taskfarm.adaptation.history.TaskFarmHistoryService;
24  import teetime.stage.taskfarm.adaptation.reconfiguration.TaskFarmReconfigurationService;
25  import teetime.stage.taskfarm.monitoring.PipeMonitoringService;
26  import teetime.stage.taskfarm.monitoring.SingleTaskFarmMonitoringService;
27  
28  /**
29   * Represents the adaptation thread used implement the self-adaptive behavior of the task farm.
30   *
31   * @author Christian Claus Wiechmann
32   *
33   * @param <I>
34   *            Input type of Task Farm
35   * @param <O>
36   *            Output type of Task Farm
37   * @param <T>
38   *            Type of the parallelized stage
39   */
40  public class AdaptationThread<I, O, T extends ITaskFarmDuplicable<I, O>> extends Thread {
41  
42  	private static final Logger LOGGER = LoggerFactory.getLogger(AdaptationThread.class);
43  
44  	private volatile boolean shouldTerminate;
45  
46  	/** task farm of this adaptation thread **/
47  	private final TaskFarmConfiguration<I, O, T> taskFarmStageConfiguration;
48  
49  	// services of this adaptation thread (history, analysis, reconfiguration)
50  	private final TaskFarmHistoryService<I, O, T> historyService;
51  	private final TaskFarmAnalysisService<I, O, T> analysisService;
52  	private final TaskFarmReconfigurationService<I, O, T> reconfigurationService;
53  
54  	private final SingleTaskFarmMonitoringService taskFarmMonitoringService;
55  	private final PipeMonitoringService pipeMonitoringService;
56  
57  	/**
58  	 * Creates an adaptation thread for the given task farm.
59  	 *
60  	 * @param taskFarmStage
61  	 *            given task farm instance
62  	 */
63  	public AdaptationThread(final DynamicTaskFarmStage<I, O, T> taskFarmStage) {
64  		this.historyService = new TaskFarmHistoryService<I, O, T>(taskFarmStage);
65  		this.analysisService = new TaskFarmAnalysisService<I, O, T>(taskFarmStage.getConfiguration());
66  		this.reconfigurationService = new TaskFarmReconfigurationService<I, O, T>(taskFarmStage);
67  		this.taskFarmStageConfiguration = taskFarmStage.getConfiguration();
68  
69  		this.taskFarmMonitoringService = new SingleTaskFarmMonitoringService(taskFarmStage, historyService);
70  		this.pipeMonitoringService = new PipeMonitoringService(historyService);
71  
72  		this.setPriority(MAX_PRIORITY);
73  	}
74  
75  	/**
76  	 * Start the execution of the adaptation thread. The execution should happen after
77  	 * the start of the merger of the corresponding task farm.
78  	 */
79  	@Override
80  	public void run() {
81  		LOGGER.debug("Adaptation thread started");
82  		while (!this.shouldTerminate) {
83  			try {
84  				executeServices();
85  				doMonitoring();
86  
87  				Thread.sleep(taskFarmStageConfiguration.getAdaptationWaitingTimeMillis());
88  			} catch (InterruptedException e) {
89  				this.shouldTerminate = true;
90  			}
91  		}
92  		LOGGER.debug("Adaptation thread stopped");
93  	}
94  
95  	private void doMonitoring() {
96  		if (this.taskFarmStageConfiguration.isMonitoringEnabled()) {
97  			this.pipeMonitoringService.doMeasurement();
98  			this.taskFarmMonitoringService.doMeasurement();
99  		}
100 	}
101 
102 	private void executeServices() throws InterruptedException {
103 		this.historyService.monitorPipes();
104 		this.analysisService.analyze(this.historyService.getHistory());
105 		this.reconfigurationService.reconfigure(this.analysisService.getThroughputScore());
106 	}
107 
108 	/**
109 	 * Terminate the adaptation thread. The termination should happen after
110 	 * the termination of the merger of the corresponding task farm.
111 	 */
112 	public void stopAdaptationThread() {
113 		this.shouldTerminate = true;
114 		interrupt();
115 		LOGGER.debug("Adaptation thread stop signal sent");
116 	}
117 
118 	/**
119 	 * Returns the {@link teetime.stage.taskfarm.adaptation.history.TaskFarmHistoryService TaskFarmHistoryService} of this adaptation thread, containing pipe
120 	 * throughput measurements.
121 	 *
122 	 * @return {@link teetime.stage.taskfarm.adaptation.history.TaskFarmHistoryService TaskFarmHistoryService} of this adaptation thread
123 	 */
124 	public TaskFarmHistoryService<I, O, T> getHistoryService() {
125 		return this.historyService;
126 	}
127 
128 	public PipeMonitoringService getPipeMonitoringService() {
129 		return pipeMonitoringService;
130 	}
131 
132 	public SingleTaskFarmMonitoringService getTaskFarmMonitoringService() {
133 		return taskFarmMonitoringService;
134 	}
135 }
