30  /**
31   * Represents the self-adaptive task farm parallelization pattern in
32   * TeeTime. It dynamically adds CPU resources at runtime depending on
33   * the current CPU load and the behavior of the parallelized stage.
34   *
35   * @author Christian Claus Wiechmann, Christoph Dornieden
36   *
37   * @param <I>
38   *            Input type of task Farm

40   *            Output type of task Farm
41   * @param <T>
42   *            Type of the parallelized stage

97  	/**
98  	 * Adds a new enclosed stage at Runtime
99  	 *
100 	 * @return new created Stage
101 	 * @throws InterruptedException
102 	 *
103 	 * @author Christian Claus Wiechmann, Christoph Dornieden (code moved from TaskFarmController)
104 	 */
105 	public ITaskFarmDuplicable<I, O> addStageAtRuntime() throws InterruptedException {
106 		if (LOGGER.isDebugEnabled()) {
107 			LOGGER.debug("Adding stage (current amount of stages: {})", getWorkerStages().size());
108 		}
109 
110 		if (!getMerger().isActive()) {
111 			getMerger().declareActive();
112 		}
113 
114 		AbstractStage basicEnclosedStage = getBasicEnclosedStage().getInputPort().getOwningStage();
115 		if (!basicEnclosedStage.isActive()) {
116 			basicEnclosedStage.declareActive();
117 		}
118 
119 		final ITaskFarmDuplicable<I, O> newStage = getBasicEnclosedStage().duplicate();
120 		// newStage.setTaskFarmStage(this);
121 
122 		final CreatePortActionDistributor<I> distributorPortAction = new CreatePortActionDistributor<I>(newStage.getInputPort(),
123 				getPipeCapacity());
124 		getDistributor().addPortActionRequest(distributorPortAction);
125 
126 		distributorPortAction.waitForCompletion();
127 
128 		final CreatePortActionMerger<O> mergerPortAction = new CreatePortActionMerger<O>(newStage.getOutputPort(),
129 				getPipeCapacity());
130 		getMerger().addPortActionRequest(mergerPortAction);
131 
132 		mergerPortAction.waitForCompletion();
133 
134 		RuntimeServiceFacade.INSTANCE.startWithinNewThread(getDistributor(), newStage.getInputPort().getOwningStage());
135 
136 		getWorkerStages().add(newStage);
137 
138 		// TODO add event "new stage added" to enable monitoring of the new pipe (see #addNewPipeToMonitoring)
139 
140 		return newStage;
141 	}

153 	/**
154 	 * Dynamically removes a stage from the controlled task farm.
155 	 *
156 	 * @throws InterruptedException
157 	 *
158 	 * @author Christian Claus Wiechmann, Christoph Dornieden (code moved from TaskFarmController)
159 	 */
160 	public ITaskFarmDuplicable<I, O> removeStageAtRuntime() throws InterruptedException {
161 		if (getWorkerStages().size() == 1) {
162 			return null;
163 		}
164 
165 		if (LOGGER.isDebugEnabled()) {
166 			LOGGER.debug("Removing stage (current amount of stages: {})", getWorkerStages().size());
167 		}
168 
169 		ITaskFarmDuplicable<I, O> stageToBeRemoved = getWorkerStages().get(getStageIndexWithLeastRemainingInput());
170 		OutputPort<? extends I> distributorOutputPort = this.getRemoveableDistributorOutputPort(stageToBeRemoved);
171 
172 		final RemovePortActionDistributor<I> distributorPortAction = new RemovePortActionDistributor<I>(distributorOutputPort);
173 		getDistributor().addPortActionRequest(distributorPortAction);
174 		getWorkerStages().remove(stageToBeRemoved);
175 
176 		distributorPortAction.waitForCompletion();
177 
178 		return stageToBeRemoved;
179 	}
180 
181 	// FIXME the task farm itself should not choose which stage to remove.
182 	// Instead, a strategy from outside the task farm should determine.
183 	private int getStageIndexWithLeastRemainingInput() {
184 		int currentMinimum = Integer.MAX_VALUE; // NOPMD (DU: caused by loop)
185 		int currentMinumumStageIndex = getWorkerStages().size() - 1;// NOPMD (DU: caused by loop)
186 
187 		// do not remove basic stage
188 		for (int i = 1; i < getWorkerStages().size(); i++) {
189 			final ITaskFarmDuplicable<I, O> instance = getWorkerStages().get(i);
190 			InputPort<I> port = instance.getInputPort();
191 			IMonitorablePipe monitorablePipe = null;
192 
193 			try {
194 				monitorablePipe = (IMonitorablePipe) port.getPipe();
195 			} catch (ClassCastException e) {
196 				throw new TaskFarmInvalidPipeException(
197 						"The input pipe of an enclosed stage instance inside a Task Farm"
198 								+ " does not implement IMonitorablePipe, which is required. Instead, the type is "
199 								+ port.getPipe().getClass().getSimpleName() + ".",
200 						e);
201 			}
202 
203 			if (monitorablePipe != null && monitorablePipe.size() < currentMinimum) {
204 				currentMinimum = monitorablePipe.size();
205 				currentMinumumStageIndex = i;
206 			}
207 		}
208 
209 		return currentMinumumStageIndex;
210 	}

212 	private OutputPort<I> getRemoveableDistributorOutputPort(final ITaskFarmDuplicable<I, O> stageToBeRemoved) {
213 		final InputPort<I> inputPortOfStage = stageToBeRemoved.getInputPort();
214 		final IPipe<?> pipeInBetween = inputPortOfStage.getPipe();
215 		@SuppressWarnings("unchecked")
216 		final OutputPort<I> distributorOutputPort = (OutputPort<I>) pipeInBetween.getSourcePort();
217 		return distributorOutputPort; // NOPMD (UnnecessaryLocalBeforeReturn: annotation should be declared at the trigger)
218 	}
