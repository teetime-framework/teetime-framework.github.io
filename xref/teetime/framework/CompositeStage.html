16  package teetime.framework;
17  
18  import teetime.framework.pipe.DummyPipe;
19  import teetime.framework.pipe.InstantiationPipe;
20  
21  /**
22   * Represents a minimal stage that composes several other stages.
23   * In order to work with this class, you need to extend from it and work from within the extending class.
24   *
25   * @since 2.0
26   *
27   * @author Christian Wulf, Nelson Tavares de Sousa
28   *
29   */
30  public class CompositeStage {
31  
32  	/**
33  	 * Default capacity for pipes
34  	 */
35  	protected static final int DEFAULT_PIPE_CAPACITY = 512;
36  
37  	protected CompositeStage() {
38  		// prohibit direct instantiation of this class
39  	}
40  
41  	/**
42  	 * Connects two ports with a pipe with a default capacity of currently {@value #DEFAULT_PIPE_CAPACITY}.
43  	 *
44  	 * @param sourcePort
45  	 *            {@link OutputPort} of the sending stage
46  	 * @param targetPort
47  	 *            {@link InputPort} of the sending stage
48  	 * @param <T>
49  	 *            the type of elements to be sent
50  	 */
51  	protected final <T> void connectPorts(final OutputPort<? extends T> sourcePort, final InputPort<T> targetPort) {
52  		connectPorts(sourcePort, targetPort, DEFAULT_PIPE_CAPACITY);
53  	}
54  
55  	/**
56  	 * Connects to ports with a pipe of a certain capacity
57  	 *
58  	 * @param sourcePort
59  	 *            {@link OutputPort} of the sending stage
60  	 * @param targetPort
61  	 *            {@link InputPort} of the sending stage
62  	 * @param capacity
63  	 *            the pipe is set to this capacity, if the value is greater than 0. If it is 0, than the pipe is unbounded, thus growing of the pipe is enabled.
64  	 * @param <T>
65  	 *            the type of elements to be sent
66  	 */
67  	protected <T> void connectPorts(final OutputPort<? extends T> sourcePort, final InputPort<T> targetPort, final int capacity) {
68  		if (sourcePort == null) {
69  			throw new IllegalArgumentException("1002 - SourcePort may not be null");
70  		}
71  		if (targetPort == null) {
72  			throw new IllegalArgumentException("1003 - TargetPort may not be null");
73  		}
74  		if (targetPort.getPipe() != null || sourcePort.getPipe() != DummyPipe.INSTANCE) {
75  			throw new IllegalStateException("1005 - Ports may not be reconnected");
76  		}
77  		new InstantiationPipe<T>(sourcePort, targetPort, capacity);
78  	}
79  
80  }
