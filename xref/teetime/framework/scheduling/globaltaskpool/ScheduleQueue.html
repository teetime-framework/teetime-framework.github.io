16  package teetime.framework.scheduling.globaltaskpool;
17  
18  import java.util.Collection;
19  
20  import teetime.framework.AbstractStage;
21  
22  /**
23   * Represents a queue which can be used by the scheduler.
24   * By definition, this data structure is ordered and potentially prioritized.
25   *
26   * @author Christian Wulf (chw)
27   *
28   * @since 3.0
29   */
30  public interface ScheduleQueue {
31  
32  	/**
33  	 * @return and removes the next stage from this queue, or <code>null</code> otherwise.
34  	 */
35  	AbstractStage removeNextStage();
36  
37  	/**
38  	 * @param stage
39  	 *            to be scheduled
40  	 * @return <code>true</code> iff the given stage could be scheduled, otherwise <code>false</code>.
41  	 */
42  	boolean scheduleStage(final AbstractStage stage);
43  
44  	/**
45  	 * @param stages
46  	 *            to be scheduled
47  	 * @return <code>true</code> iff all of the given stages could be scheduled, otherwise <code>false</code>.
48  	 */
49  	default boolean scheduleStages(final Collection<? extends AbstractStage> stages) {
50  		boolean scheduledAllStages = true;
51  		for (AbstractStage stage : stages) {
52  			boolean scheduledStage = scheduleStage(stage);
53  			scheduledAllStages = scheduledAllStages && scheduledStage;
54  		}
55  		return scheduledAllStages;
56  	}
57  
58  }
