1   package teetime.framework.scheduling.globaltaskpool;
2   
3   import java.util.Collection;
4   
5   import teetime.framework.AbstractStage;
6   
7   /**
8    * Represents a queue which can be used by the scheduler.
9    * By definition, this data structure is ordered and potentially prioritized.
10   *
11   * @author Christian Wulf (chw)
12   *
13   * @since 3.0
14   */
15  public interface ScheduleQueue {
16  
17  	/**
18  	 * @return and removes the next stage from this queue, or <code>null</code> otherwise.
19  	 */
20  	AbstractStage removeNextStage();
21  
22  	/**
23  	 * @param stage
24  	 *            to be scheduled
25  	 * @return <code>true</code> iff the given stage could be scheduled, otherwise <code>false</code>.
26  	 */
27  	boolean scheduleStage(final AbstractStage stage);
28  
29  	/**
30  	 * @param stages
31  	 *            to be scheduled
32  	 * @return <code>true</code> iff all of the given stages could be scheduled, otherwise <code>false</code>.
33  	 */
34  	default boolean scheduleStages(final Collection<? extends AbstractStage> stages) {
35  		boolean scheduledAllStages = true;
36  		for (AbstractStage stage : stages) {
37  			boolean scheduledStage = scheduleStage(stage);
38  			scheduledAllStages = scheduledAllStages && scheduledStage;
39  		}
40  		return scheduledAllStages;
41  	}
42  
43  }
