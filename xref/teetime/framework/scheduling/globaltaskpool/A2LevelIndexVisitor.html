1   package teetime.framework.scheduling.globaltaskpool;
2   
3   import teetime.framework.AbstractPort;
4   import teetime.framework.AbstractStage;
5   import teetime.framework.ITraverserVisitor;
6   import teetime.framework.StageFacade;
7   import teetime.framework.Traverser.VisitorBehavior;
8   import teetime.framework.pipe.DummyPipe;
9   
10  class A2LevelIndexVisitor implements ITraverserVisitor {
11  
12  	private static final StageFacade STAGE_FACADE = StageFacade.INSTANCE;
13  
14  	private int maxLevelIndex;
15  
16  	@Override
17  	public VisitorBehavior visit(final AbstractStage stage) {
18  		return VisitorBehavior.CONTINUE_FORWARD;
19  	}
20  
21  	@Override
22  	public VisitorBehavior visit(final AbstractPort<?> port) {
23  		AbstractStage sourceStage = port.getOwningStage();
24  		AbstractStage targetStage = port.getPipe().getTargetPort().getOwningStage();
25  
26  		int targetLevelIndex = STAGE_FACADE.getLevelIndex(targetStage);
27  		int sourceLevelIndex = STAGE_FACADE.getLevelIndex(sourceStage);
28  		int levelIndex = Math.max(targetLevelIndex, sourceLevelIndex + 1);
29  		STAGE_FACADE.setLevelIndex(targetStage, levelIndex);
30  
31  		maxLevelIndex = Math.max(maxLevelIndex, levelIndex);
32  
33  		return VisitorBehavior.CONTINUE_FORWARD;
34  	}
35  
36  	@Override
37  	public void visit(final DummyPipe pipe, final AbstractPort<?> port) {
38  		// do nothing
39  	}
40  
41  	public int getMaxLevelIndex() {
42  		return maxLevelIndex;
43  	}
44  
45  }
