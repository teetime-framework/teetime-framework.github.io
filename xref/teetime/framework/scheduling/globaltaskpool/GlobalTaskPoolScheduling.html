37  /**
38   * This scheduling approach maintains a global, synchronized, prioritized task pool whose tasks are stages.
39   * Multiple threads access this task pool concurrently.
40   * At each moment in time, a particular stage is executed only by at most one thread.
41   * Thus, a stage in combination with the task data structure acts as a lock for executing that stage.
42   *
43   * @author Christian Wulf (chw)
44   *
45   * @since 3.0
46   *
47   */
48  public class GlobalTaskPoolScheduling implements TeeTimeScheduler, PipeScheduler, UncaughtExceptionHandler {

53  	private static final int DEFAULT_NUM_OF_EXECUTIONS = 1;

58  	private final Set<AbstractStage> frontStages = ConcurrentHashMap.newKeySet();

88  	/**
89  	 * @param numThreads
90  	 *            the number of threads to use for executing the given P&amp;F configuration
91  	 * @param configuration
92  	 *            the configuration to execute/schedule
93  	 * @param numOfExecutions
94  	 *            the number of execution per scheduled stage (task) for a thread. Is rounded up to the next power of 2, i.e., <code>1,2,4,16,...</code>
95  	 */
96  	public GlobalTaskPoolScheduling(final int numThreads, final Configuration configuration, final int numOfExecutions) {
97  		this.numThreads = numThreads;
98  		this.configuration = configuration;
99  		if (numOfExecutions <= 0) {
100 			throw new IllegalArgumentException("numOfExecutions is " + numOfExecutions + ", but must have a positive value.");
101 		}
102 		int actualNumOfExecutions = Pow2.roundToPowerOfTwo(numOfExecutions);
103 		this.actualNumOfExecutions = actualNumOfExecutions;
104 		this.numOfExecutionsMask = actualNumOfExecutions - 1;
105 	}

113 	@Override
114 	public void onInitialize() {
115 		Collection<AbstractStage> startStages = CONFIG_FACADE.getStartStages(configuration);
116 		initialize(startStages);
117 
118 		initializeThreads(numThreads, regularThreads, "regular");
119 		// We need at most n-1 backup threads.
120 		// Consider the situation where all consumers are slower than their producers.
121 		// Then, all stages but sinks are paused.
122 		// For each of these paused stage, the scheduler requires a backup thread.
123 		initializeThreads(allStages.size() - 1, backupThreads, "backup");
124 	}
125 
126 	private void initializeThreads(final int size, final List<TeeTimeTaskQueueThreadChw> threads, final String threadNameSuffix) {
127 		final AbstractExceptionListenerFactory<?> factory = ConfigurationFacade.INSTANCE.getFactory(configuration);
128 
129 		for (int i = 0; i < size; i++) {
130 			TeeTimeTaskQueueThreadChw backupThread = new TeeTimeTaskQueueThreadChw(this, actualNumOfExecutions);
131 			backupThread.setName(backupThread.getName() + "-" + threadNameSuffix);
132 			AbstractExceptionListener listener = factory.createInstance(backupThread);
133 			backupThread.setExceptionListener(listener);
134 			backupThread.setUncaughtExceptionHandler(this);
135 			backupThread.start();
136 
137 			threads.add(backupThread);
138 		}
139 	}

141 	private void initialize(final Collection<AbstractStage> startStages) {
142 		// TODO: Add port type validation again.
143 		if (startStages.isEmpty()) {
144 			throw new IllegalStateException("No start stages passed. You need to pass at least one start stage.");
145 		}
146 
147 		A1CreatedStageCollector stageCollector = new A1CreatedStageCollector();
148 		Traverser traversor = new Traverser(stageCollector);
149 		for (AbstractStage startStage : startStages) {
150 			traversor.traverse(startStage);
151 		}
152 
153 		allStages = stageCollector.getStages();
154 
155 		for (AbstractStage stage : allStages) {
156 			categorizeStage(stage);
157 			setScheduler(stage);
158 		}
159 
160 		if (finiteProducerStages.isEmpty()) {
161 			throw new IllegalStateException("1004 - No producer stages in this configuration.");
162 		}
163 
164 		// (re-)compute level index for each stage
165 		A2LevelIndexVisitor levelIndexVisitor = new A2LevelIndexVisitor();
166 		traversor = new Traverser(levelIndexVisitor);
167 		for (AbstractStage startStage : finiteProducerStages) {
168 			traversor.traverse(startStage);
169 		}
170 
171 		final int capacity = numRunningStages.getCurrentCount();
172 		taskPool = new PrioritizedTaskPool(levelIndexVisitor.getMaxLevelIndex() + 1, capacity);
173 		taskPool.scheduleStages(frontStages);
174 
175 		// instantiate pipes
176 		int requestPipeCapcity = actualNumOfExecutions * 128; // with additional buffer factor
177 		A3PipeInstantiation pipeVisitor = new A3PipeInstantiation(this, requestPipeCapcity);
178 		traversor = new Traverser(pipeVisitor);
179 		for (AbstractStage startStage : startStages) {
180 			traversor.traverse(startStage);
181 		}
182 	}
183 
184 	private void categorizeStage(final AbstractStage stage) {
185 		switch (STAGE_FACADE.getTerminationStrategy(stage)) {
186 		case BY_INTERRUPT:
187 			throw new IllegalStateException("Infinite producers are not supported by this scheduling strategy.");
188 		case BY_SELF_DECISION:
189 			finiteProducerStages.add(stage);
190 			frontStages.add(stage);
191 			numRunningStages.countUp();
192 			break;
193 		case BY_SIGNAL:
194 			numRunningStages.countUp();
195 			break;
196 		default:
197 			LOGGER.warn("Unknown termination strategy '{}' in stage {}", STAGE_FACADE.getTerminationStrategy(stage), stage);
198 			break;
199 		}
200 	}

202 	@Override
203 	public void onValidate() {
204 		// // BETTER validate concurrently
205 		final ValidatingSignal signal = new ValidatingSignal();
206 		SignalVisitor signalVisitor = new SignalVisitor(signal);
207 		BreadthFirstTraverser traversor = new BreadthFirstTraverser();
208 
209 		synchronized (frontStages) {
210 			for (AbstractStage stage : frontStages) {
211 				traversor.traverse(stage, signalVisitor);
212 
213 				if (signal.getInvalidPortConnections().size() > 0) {
214 					throw new AnalysisNotValidException(signal.getInvalidPortConnections());
215 				}
216 			}
217 		}
218 	}

220 	@Override
221 	public void onExecute() {
222 		final StartingSignal signal = new StartingSignal();
223 		SignalVisitor signalVisitor = new SignalVisitor(signal);
224 		BreadthFirstTraverser traversor = new BreadthFirstTraverser();
225 
226 		synchronized (frontStages) {
227 			for (AbstractStage stage : frontStages) {
228 				traversor.traverse(stage, signalVisitor);
229 			}
230 		}
231 
232 		// TODO move before onExecute so that starting the threads does not count to the execution time #350
233 		for (TeeTimeTaskQueueThreadChw thread : regularThreads) {
234 			thread.awake();
235 		}
236 	}
237 
238 	@Override
239 	public void onTerminate() {
240 		synchronized (frontStages) {
241 			for (AbstractStage finiteProducerStage : frontStages) {
242 				STAGE_FACADE.abort(finiteProducerStage);
243 			}
244 		}
245 	}
246 
247 	@Override
248 	public void onFinish() {
249 		try {
250 			numRunningStages.await();
251 			LOGGER.debug("Finished execution.");
252 		} catch (InterruptedException e) {
253 			LOGGER.warn("Interrupted execution.", e);
254 		}
255 
256 		try {
257 			for (TeeTimeTaskQueueThreadChw thread : regularThreads) {
258 				thread.awake();
259 				thread.join();
260 			}
261 			for (TeeTimeTaskQueueThreadChw thread : backupThreads) {
262 				thread.awake();
263 				thread.join();
264 			}
265 		} catch (InterruptedException e) {
266 			throw new IllegalStateException(e);
267 		}
268

270 		// if (!exceptions.isEmpty()) {
271 		// throw new ExecutionException(exceptions);
272 		// }
273 	}

328 	@Override
329 	public void onElementAdded(final AbstractSynchedPipe<?> pipe) {
330 		BoundedMpMcSynchedPipe<?> castedPipe = (BoundedMpMcSynchedPipe<?>) pipe;
331 		long numPushes = castedPipe.getNumPushesSinceAppStart();
332 		long lastNumPushes = castedPipe.getLastProducerIndex();
333 		// performance optimization: & represents % (modulo)
334 		// if ((numPushes & numOfExecutionsMask) != 0) {
335 		if (numPushes - lastNumPushes >= actualNumOfExecutions) {
336 			castedPipe.setLastProducerIndex(numPushes);
337 			AbstractStage targetStage = pipe.getCachedTargetStage();
338 			// We do not schedule the target stage on each incoming element.
339 			// Instead, we schedule it after adding 'actualNumOfExecutions' elements.
340 			if (!taskPool.scheduleStage(targetStage)) {
341 				String message = String.format("Could not schedule %s; pool=%s", targetStage, taskPool);
342 				throw new IllegalStateException(message);
343 			}
344 		}
345 	}
346 
347 	@Override
348 	public void onElementNotAdded(final AbstractSynchedPipe<?> pipe) {
349 		if (!taskPool.scheduleStage(pipe.getCachedTargetStage())) {
350 			throw new IllegalStateException(String.format("onElementNotAdded: scheduling target stage failed for %s", pipe.getCachedTargetStage()));
351 		}
352 
353 		AbstractStage owningStage = pipe.getSourcePort().getOwningStage();
354 		LoggerFactory.getLogger(owningStage.getClass()).debug("Yielding {} cause of the full pipe {}", owningStage, this);
355 		this.yieldStage(owningStage);
356 	}

358 	/**
359 	 * Among others, pauses the executing thread.
360 	 *
361 	 * @param stage
362 	 */
363 	private void yieldStage(final AbstractStage stage) {
364 		// 1. update state attributes before scheduling the stage again
365 		stage.setPaused(true);
366 		// allow other to execute the stage (only) in order to awake the current thread again
367 		setIsBeingExecuted(stage, false);
368 
369 		if (!taskPool.scheduleStage(stage)) {
370 			throw new IllegalStateException(String.format("(yieldStage) Self-scheduling failed for %s", stage));
371 		}
372 
373 		// 2. awake any backup thread after scheduling the stage
374 		backupThreads.remove(0).awake();
375 
376 		getCurrentThread().pause();
377 
378 		if (!stage.isBeingExecuted()) {
379 			throw new IllegalStateException("Stage must be in state 'is being executed'");
380 		}
381 
382 		stage.setPaused(false);
383 		LOGGER.debug("Continue with {}", stage);
384 	}

390 	/**
391 	 * Revokes the stage's pause and pauses the current thread afterwards.
392 	 *
393 	 * @param stage
394 	 */
395 	void continueStage(final AbstractStage stage) {
396 		TeeTimeTaskQueueThreadChw thisThread = getCurrentThread();
397 		backupThreads.add(thisThread);
398 
399 		if (!stage.isBeingExecuted()) {
400 			throw new IllegalStateException("Stage must be in state 'is being executed'");
401 		}
402 
403 		/* must follow "backupThreads.add" so that the awakened thread can invoke "backupThreads.remove(0)" without causing an IndexOutOfBoundsException */
404 		TeeTimeTaskQueueThreadChw owningThread = this.getOwningThreadSynched(stage);
405 		owningThread.awake();
406 
407 		thisThread.pause();
408 		LOGGER.debug("Continue (backup) with {}", stage);
409 	}
