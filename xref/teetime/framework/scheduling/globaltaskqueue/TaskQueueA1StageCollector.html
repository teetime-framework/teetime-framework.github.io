12  class TaskQueueA1StageCollector implements ITraverserVisitor {
13  
14  	private final Set<AbstractStage> stages = new HashSet<AbstractStage>();
15  
16  	public Set<AbstractStage> getStages() {
17  		return this.stages;
18  	}
19  
20  	@Override
21  	public Traverser.VisitorBehavior visit(final AbstractStage stage) {
22  		if (!stages.contains(stage) && stage.getCurrentState() == StageState.CREATED) {
23  			stages.add(stage);
24  		}
25  		// visitor termination condition: stop if the stage already runs or has been terminated
26  		return stage.getCurrentState() == StageState.CREATED ? Traverser.VisitorBehavior.CONTINUE : Traverser.VisitorBehavior.STOP;
27  	}
28  
29  	@Override
30  	public Traverser.VisitorBehavior visit(final AbstractPort<?> port) {
31  		return Traverser.VisitorBehavior.CONTINUE;
32  	}
33  
34  	@Override
35  	public void visit(final DummyPipe pipe, final AbstractPort<?> port) {
36  		// do nothing
37  	}
38  
39  }
