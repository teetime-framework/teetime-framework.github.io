16  package teetime.framework.scheduling.globaltaskqueue;
17  
18  import java.util.HashSet;
19  import java.util.Set;
20  
21  import teetime.framework.*;
22  import teetime.framework.pipe.DummyPipe;
23  
24  /**
25   * Created by nilsziermann on 30.12.16.
26   */
27  class TaskQueueA1StageCollector implements ITraverserVisitor {
28  
29  	private final Set<AbstractStage> stages = new HashSet<AbstractStage>();
30  
31  	public Set<AbstractStage> getStages() {
32  		return this.stages;
33  	}
34  
35  	@Override
36  	public Traverser.VisitorBehavior visit(final AbstractStage stage) {
37  		if (!stages.contains(stage) && stage.getCurrentState() == StageState.CREATED) {
38  			stages.add(stage);
39  		}
40  		// visitor termination condition: stop if the stage already runs or has been terminated
41  		return stage.getCurrentState() == StageState.CREATED ? Traverser.VisitorBehavior.CONTINUE_BACK_AND_FORTH : Traverser.VisitorBehavior.STOP;
42  	}
43  
44  	@Override
45  	public Traverser.VisitorBehavior visit(final AbstractPort<?> port) {
46  		return Traverser.VisitorBehavior.CONTINUE_BACK_AND_FORTH;
47  	}
48  
49  	@Override
50  	public void visit(final DummyPipe pipe, final AbstractPort<?> port) {
51  		// do nothing
52  	}
53  
54  }
