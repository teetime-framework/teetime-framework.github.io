16  package teetime.framework.scheduling.globaltaskqueue;
17  
18  import java.util.List;
19  
20  import teetime.framework.*;
21  import teetime.framework.signal.ISignal;
22  import teetime.framework.signal.TerminatingSignal;
23  
24  public class TeeTimeTaskQueueThreadChw extends Thread {
25  
26  	private static final StageFacade STAGE_FACADE = StageFacade.INSTANCE;
27  
28  	private final GlobalTaskQueueScheduling scheduling;
29  	private final int numOfExecutions;
30  
31  	public TeeTimeTaskQueueThreadChw(final GlobalTaskQueueScheduling scheduling, final int numOfExecutions) {
32  		super();
33  		this.scheduling = scheduling;
34  		this.numOfExecutions = numOfExecutions;
35  	}
36  
37  	@Override
38  	public void run() {
39  		final TaskQueue taskQueue = scheduling.getTaskQueue(); // NOPMD (DU anomaly)
40  
41  		// TODO implement: run as long as there is at least one stage which has not been terminated yet
42  		while (scheduling.getNumNonTerminatedFiniteStages().get() > 0) {
43  			AbstractStage stage = taskQueue.removeNextStage();
44  			if (stage != null) {
45  				try {
46  					executeStage(stage);
47  					refillTaskQueue(stage, taskQueue);
48  				} finally {
49  					taskQueue.releaseStage(stage); // release lock (FIXME bad API)
50  				}
51  			}
52  		}
53  	}
54  
55  	private void executeStage(final AbstractStage stage) {
56  		STAGE_FACADE.runStage(stage, numOfExecutions);
57  
58  		if (STAGE_FACADE.shouldBeTerminated(stage)) {
59  			afterStageExecution(stage);
60  		}
61  
62  		if (stage.getCurrentState() == StageState.TERMINATED) {
63  			// int newValue =
64  			scheduling.getNumNonTerminatedFiniteStages().decrementAndGet();
65  
66  			// FIXME so far, it is unclear to me when to re-add infinite producers
67  			// => solution: remove the concept of infinite producers
68  
69  			passFrontStatusToSuccessorStages(stage);
70  		}
71  	}
72  
73  	private void afterStageExecution(final AbstractStage stage) {
74  		if (stage instanceof AbstractProducerStage) {
75  			stage.onSignal(new TerminatingSignal(), null);
76  		} else if (stage instanceof AbstractConsumerStage) {
77  			final ISignal signal = new TerminatingSignal(); // NOPMD DU caused by loop
78  			for (InputPort<?> inputPort : STAGE_FACADE.getInputPorts(stage)) {
79  				stage.onSignal(signal, inputPort);
80  			}
81  		}
82  	}
83  
84  	private void passFrontStatusToSuccessorStages(final AbstractStage stage) {
85  		List<AbstractStage> frontStages = scheduling.getFrontStages();
86  		synchronized (frontStages) {
87  			if (frontStages.contains(stage)) {
88  				frontStages.remove(stage);
89  				for (OutputPort<?> outputPort : STAGE_FACADE.getOutputPorts(stage)) {
90  					AbstractStage targetStage = outputPort.getPipe().getTargetPort().getOwningStage();
91  					if (!STAGE_FACADE.shouldBeTerminated(targetStage) && targetStage.getCurrentState() != StageState.TERMINATED) {
92  						frontStages.add(targetStage);
93  					}
94  				}
95  			}
96  		}
97  	}
98  
99  	private void refillTaskQueue(final AbstractStage stage, final TaskQueue taskQueue) {
100 		// Add all successor stages so that they will be executed afterwards.
101 		// TODO evaluate whether adding workless stages is faster than adding stages within pipes.
102 		List<OutputPort<?>> outputPorts = STAGE_FACADE.getOutputPorts(stage);
103 		for (OutputPort<?> outputPort : outputPorts) {
104 			AbstractStage targetStage = outputPort.getPipe().getTargetPort().getOwningStage();
105 			if (!STAGE_FACADE.shouldBeTerminated(targetStage) && targetStage.getCurrentState() != StageState.TERMINATED) {
106 				taskQueue.scheduleStage(targetStage);
107 			}
108 		}
109 
110 		// re-add stage to task queue if it is a front stage (a terminated front stage would have been already removed at this point)
111 		if (scheduling.getFrontStages().contains(stage)) {
112 			taskQueue.scheduleStage(stage);
113 		}
114 	}
115 }
