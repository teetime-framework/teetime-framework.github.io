11  public class TeeTimeTaskQueueThreadChw extends Thread {
12  
13  	private static final StageFacade STAGE_FACADE = StageFacade.INSTANCE;
14  
15  	private static final AbstractStage TERM_STAGE = new AbstractStage() {
16  		@Override
17  		protected void execute() throws Exception {// do nothing
18  		}
19  	};
20  
21  	private final GlobalTaskQueueScheduling scheduling;
22  	private final MpmcArrayQueue<AbstractStage> taskQueue;
23  
24  	public TeeTimeTaskQueueThreadChw(final GlobalTaskQueueScheduling scheduling) {
25  		super();
26  		this.scheduling = scheduling;
27  		this.taskQueue = scheduling.getTaskQueue();
28  	}
29  
30  	@Override
31  	public void run() {
32  		while (true) {
33  			AbstractStage stage = taskQueue.poll();
34  			if (stage == TERM_STAGE) {
35  				// TODO terminate thread
36  			} else if (stage != null) {
37  				executeStage(stage);
38  			} else {
39  				// boolean finiteProducerStagesRunning = taskQueue.addAll(scheduling.getFiniteProducerStages());
40  
41  				// if (finiteProducerStagesRunning) {
42  				// taskQueue.addAll(scheduling.getInfiniteProducerStages());
43  				// } else {
44  				// break;
45  				// }
46  			}
47  		}
48  	}
49  
50  	private void executeStage(final AbstractStage stage) {
51  		int numOfExecutions = 1;
52  
53  		STAGE_FACADE.runStage(stage, numOfExecutions);
54  
55  		// Add all successor stages so that they will be executed afterwards.
56  		// TODO evaluate whether adding workless stages is faster than adding stages within pipes.
57  		List<OutputPort<?>> outputPorts = STAGE_FACADE.getOutputPorts(stage);
58  		for (OutputPort<?> outputPort : outputPorts) {
59  			AbstractStage targetStage = outputPort.getPipe().getTargetPort().getOwningStage();
60  			if (!STAGE_FACADE.shouldBeTerminated(targetStage) && targetStage.getCurrentState() != StageState.TERMINATED) {
61  				taskQueue.add(targetStage);
62  			}
63  		}
64  
65  		if (STAGE_FACADE.shouldBeTerminated(stage)) {
66  			afterStageExecution(stage);
67  		}
68  	}
69  
70  	private void afterStageExecution(final AbstractStage stage) {
71  		if (stage instanceof AbstractProducerStage) {
72  			stage.onSignal(new TerminatingSignal(), null);
73  		} else if (stage instanceof AbstractConsumerStage) {
74  			final ISignal signal = new TerminatingSignal(); // NOPMD DU caused by loop
75  			for (InputPort<?> inputPort : STAGE_FACADE.getInputPorts(stage)) {
76  				stage.onSignal(signal, inputPort);
77  			}
78  		}
79  	}
80  }
