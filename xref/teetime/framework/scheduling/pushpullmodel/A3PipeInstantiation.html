16  package teetime.framework.scheduling.pushpullmodel;
17  
18  import java.util.HashSet;
19  import java.util.Set;
20  
21  import org.slf4j.Logger;
22  import org.slf4j.LoggerFactory;
23  
24  import teetime.framework.AbstractPort;
25  import teetime.framework.AbstractStage;
26  import teetime.framework.ITraverserVisitor;
27  import teetime.framework.Traverser.VisitorBehavior;
28  import teetime.framework.pipe.*;
29  import teetime.framework.scheduling.PipeScheduler;
30  
31  /**
32   * Automatically instantiates the correct pipes
33   */
34  class A3PipeInstantiation implements ITraverserVisitor, PipeScheduler {
35  
36  	private static final Logger LOGGER = LoggerFactory.getLogger(A3PipeInstantiation.class);
37  
38  	private final Set<IPipe<?>> visitedPipes = new HashSet<IPipe<?>>();
39  
40  	@Override
41  	public VisitorBehavior visit(final AbstractStage stage) {
42  		return VisitorBehavior.CONTINUE_BACK_AND_FORTH;
43  	}
44  
45  	@Override
46  	public VisitorBehavior visit(final AbstractPort<?> port) {
47  		IPipe<?> pipe = port.getPipe();
48  		if (visitedPipes.contains(pipe)) {
49  			return VisitorBehavior.STOP; // NOPMD two returns are better
50  		}
51  		visitedPipes.add(pipe);
52  
53  		instantiatePipe(pipe);
54  
55  		return VisitorBehavior.CONTINUE_BACK_AND_FORTH;
56  	}
57  
58  	@Override
59  	public void visit(final DummyPipe pipe, final AbstractPort<?> port) {
60  		if (LOGGER.isInfoEnabled()) {
61  			LOGGER.info("Unconnected port {} in stage {}", port, port.getOwningStage().getId());
62  		}
63  	}
64  
65  	private <T> void instantiatePipe(final IPipe<T> pipe) {
66  		IPipe<T> instantiatedPipe;
67  
68  		AbstractStage sourceStage = pipe.getSourcePort().getOwningStage();
69  		AbstractStage targetStage = pipe.getTargetPort().getOwningStage();
70  
71  		if (!(pipe instanceof InstantiationPipe)) { // if manually connected
72  			instantiatedPipe = pipe;
73  		} else if (!targetStage.isActive() || sourceStage == targetStage) { // NOPMD .equals() can't be used here
74  			// normal or reflexive pipe => intra
75  			instantiatedPipe = new UnsynchedPipe<T>(pipe.getSourcePort(), pipe.getTargetPort());
76  			LOGGER.debug("Connected (unsynch) {} and {}", pipe.getSourcePort(), pipe.getTargetPort());
77  		} else if (pipe.capacity() == 0) {
78  			// synchronized, unlimited capacity
79  			instantiatedPipe = new UnboundedSynchedPipe<T>(pipe.getSourcePort(), pipe.getTargetPort());
80  			LOGGER.debug("Connected (unbounded) {} and {}", pipe.getSourcePort(), pipe.getTargetPort());
81  		} else {
82  			// synchronized, limited capacity
83  			instantiatedPipe = new BoundedSynchedPipe<T>(pipe.getSourcePort(), pipe.getTargetPort(), pipe.capacity());
84  			LOGGER.debug("Connected (bounded) {} and {}", pipe.getSourcePort(), pipe.getTargetPort());
85  		}
86  
87  		instantiatedPipe.setScheduler(this);
88  	}
89  
90  	@Override
91  	public void onElementAdded(final AbstractUnsynchedPipe<?> pipe) {
92  		pipe.getCachedTargetStage().executeByFramework();
93  	}
94  
95  	@Override
96  	public void onElementAdded(final AbstractSynchedPipe<?> pipe) {
97  		// do nothing
98  	}
99  
100 	@Override
101 	public void onElementNotAdded(final AbstractSynchedPipe<?> pipe) {
102 		// do nothing
103 	}
104 
105 }
