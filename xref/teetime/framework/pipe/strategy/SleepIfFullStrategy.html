3   import teetime.framework.StageState;
4   import teetime.framework.exceptionHandling.TerminateException;
5   import teetime.framework.pipe.IPipe;
6   
7   public class SleepIfFullStrategy implements PipeElementInsertionStrategy {
8   
9   	// statistics
10  	private int numWaits;
11  
12  	@Override
13  	public boolean add(final IPipe<?> pipe, final Object element) {
14  		while (!pipe.addNonBlocking(element)) {
15  			// the following sending*-related lines are commented out since they are computationally too expensive
16  			// this.getSourcePort().getOwningStage().sendingFailed();
17  			// Thread.yield();
18  			StageState targetStageState = pipe.getTargetPort().getOwningStage().getCurrentState();
19  			if (targetStageState == StageState.TERMINATED ||
20  					Thread.currentThread().isInterrupted()) {
21  				throw TerminateException.INSTANCE;
22  			}
23  			this.numWaits++;
24  			try {
25  				Thread.sleep(10);
26  			} catch (InterruptedException ignore) { // NOPMD can be interrupted w/o any reason
27  				throw TerminateException.INSTANCE;
28  			}
29  		}
30  		// this.getSourcePort().getOwningStage().sendingSucceeded();
31  		// this.reportNewElement();
32  		return true;
33  	}
34  
35  	public int getNumWaits() {
36  		return numWaits;
37  	}
38  
39  }
