16  package teetime.framework;
17  
18  import java.util.HashSet;
19  import java.util.Set;
20  
21  import teetime.framework.Traverser.VisitorBehavior;
22  import teetime.framework.pipe.DummyPipe;
23  
24  /**
25   * Searches for threadable stages
26   */
27  class A1ThreadableStageCollector implements ITraverserVisitor {
28  
29  	private final Set<AbstractStage> threadableStages = new HashSet<AbstractStage>();
30  
31  	public Set<AbstractStage> getThreadableStages() {
32  		return threadableStages;
33  	}
34  
35  	@Override
36  	public VisitorBehavior visit(final AbstractStage stage) {
37  		if (stage.getInputPorts().size() == 0) { // activate producers automatically
38  			stage.declareActive();
39  		}
40  
41  		if (stage.isActive() && !threadableStages.contains(stage)) {
42  			threadableStages.add(stage);
43  		}
44  
45  		return VisitorBehavior.CONTINUE;
46  	}
47  
48  	@Override
49  	public VisitorBehavior visit(final AbstractPort<?> port) {
50  		return VisitorBehavior.CONTINUE;
51  	}
52  
53  	@Override
54  	public void visit(final DummyPipe pipe, final AbstractPort<?> port) {
55  		// do nothing
56  	}
57  
58  }
