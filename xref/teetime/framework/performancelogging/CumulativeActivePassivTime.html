16  package teetime.framework.performancelogging;
17  
18  class CumulativeActivePassivTime implements ActivationStateLogger.IFormatingStrategy {
19  
20  	/**
21  	 *
22  	 */
23  	private final ActivationStateLogger stateLogger;
24  
25  	/**
26  	 * @param stateLogger
27  	 */
28  	CumulativeActivePassivTime(final ActivationStateLogger stateLogger) {
29  		this.stateLogger = stateLogger;
30  	}
31  
32  	@Override
33  	public String formatData() {
34  		String result = "\n Formating to analyse differences in runtime:\n\n";
35  
36  		result += "name;cumulative active time;cumulative active waiting time;cumulative blocked time; total time\n";
37  
38  		// go through all the stages
39  		for (StateLoggable stage : this.stateLogger.getStages()) {
40  			// first add a formated version of their names to the line.
41  			result += this.stateLogger.formateName(stage);
42  
43  			long earliestTimeStamp = stage.getStates().get(0).getTimeStamp();
44  			long latestTimeStamp = earliestTimeStamp;
45  			long lastTimeStamp = earliestTimeStamp;
46  			long cumulativeActiveTime = 0;
47  			long cumulativeActiveWaitingTime = 0;
48  			long cumulativeBlockedTime = 0;
49  
50  			// go through all states of this stage and sum up the active times while counting the number of active times
51  			for (StateChange state : stage.getStates()) {
52  				long actualTimeStamp = state.getTimeStamp();
53  
54  				if (actualTimeStamp < earliestTimeStamp) {
55  					earliestTimeStamp = actualTimeStamp;
56  					System.err.println("Some later State has a lower timestamp value!");
57  				} else if (actualTimeStamp > latestTimeStamp) {
58  					latestTimeStamp = actualTimeStamp;
59  				}
60  
61  				long elapsedTime = actualTimeStamp - lastTimeStamp;
62  
63  				switch (state.getExecutionState()) {
64  				case ACTIVE:
65  					cumulativeActiveTime += elapsedTime;
66  					break;
67  				case ACTIVE_WAITING:
68  					cumulativeActiveWaitingTime += elapsedTime;
69  					break;
70  				case BLOCKED:
71  					cumulativeBlockedTime += elapsedTime;
72  					break;
73  				default:
74  					break;
75  				}
76  
77  				lastTimeStamp = actualTimeStamp;
78  			}
79  
80  			result += cumulativeActiveTime + ";" + cumulativeActiveWaitingTime + ";" + cumulativeBlockedTime + ";" + (latestTimeStamp - earliestTimeStamp);
81  
82  			result += "\n";
83  		}
84  		return result.replace("\n", String.format("%n"));
85  	}
86  
87  }
