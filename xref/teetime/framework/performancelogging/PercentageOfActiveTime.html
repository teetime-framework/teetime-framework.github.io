3   class PercentageOfActiveTime implements ActivationStateLogger.IFormatingStrategy {
4   
5   	/**
6   	 *
7   	 */
8   	private final ActivationStateLogger stateLogger;
9   
10  	/**
11  	 * @param stateLogger
12  	 */
13  	PercentageOfActiveTime(final ActivationStateLogger stateLogger) {
14  		this.stateLogger = stateLogger;
15  	}
16  
17  	@Override
18  	public String formatData() {
19  		String result = "\n	Formating of the data to get Percentage of active time:\n\n";
20  
21  		for (StateLoggable stage : this.stateLogger.getStages()) {
22  			result += this.stateLogger.formateName(stage);
23  
24  			long firstTimeStamp = stage.getStates().get(0).getTimeStamp();
25  			long lastTimeStamp = Long.MIN_VALUE;
26  			long commulativeActiveTime = 0;
27  			long lastActiveTimeStamp = 0;
28  			boolean lastActive = false;
29  
30  			for (ActivationState state : stage.getStates()) {
31  				long currentTimeStamp = state.getTimeStamp();
32  				if (state.getCause() == ActivationState.NOTHING_FAILED) {
33  					if (!lastActive) {
34  						lastActiveTimeStamp = currentTimeStamp;
35  					}
36  					lastActive = true;
37  				} else {
38  					if (lastActive && lastActiveTimeStamp != 0) {
39  						commulativeActiveTime += (currentTimeStamp - lastActiveTimeStamp);
40  					}
41  					lastActive = false;
42  				}
43  				if (currentTimeStamp > lastTimeStamp) {
44  					lastTimeStamp = currentTimeStamp;
45  				}
46  				if (currentTimeStamp < firstTimeStamp) {
47  					firstTimeStamp = currentTimeStamp;
48  				}
49  			}
50  
51  			long totalTime = (lastActiveTimeStamp - firstTimeStamp);
52  
53  			// Add formated data to the line
54  			result += " Percentage of ActiveTime: "
55  					// I differentiate between stages that were active the whole time and the ones that were interrupted in between.
56  					// this will help to keep track of the necessary information.
57  					+ ((totalTime != 0) ? (commulativeActiveTime / (lastActiveTimeStamp - firstTimeStamp)) : "100") + "%";
58  
59  			result += "\n";
60  		}
61  		return result.replace("\n", String.format("%n"));
62  	}
63  
64  }
