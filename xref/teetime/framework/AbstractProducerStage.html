16  package teetime.framework;
17  
18  import org.slf4j.Logger;
19  
20  /**
21   * The <code>ProducerStage</code> produces at least one element at each execution.<br>
22   *
23   * @author Christian Wulf
24   *
25   * @param <O>
26   *            the type of the default output port
27   *
28   */
29  public abstract class AbstractProducerStage<O> extends AbstractStage {
30  
31  	protected final OutputPort<O> outputPort = this.createOutputPort();
32  
33  	public AbstractProducerStage() {
34  		super();
35  	}
36  
37  	/**
38  	 * @param logger
39  	 *            a custom logger (potentially shared by multiple stage instances)
40  	 */
41  	public AbstractProducerStage(final Logger logger) {
42  		super(logger);
43  	}
44  
45  	public final OutputPort<O> getOutputPort() {
46  		return this.outputPort;
47  	}
48  
49  	/**
50  	 * @deprecated since 3.0.
51  	 *             We will completely remove framework-backed support for infinite producers since it has never worked correctly in all (corner) cases.
52  	 *             Instead, please use finite producers and implement an appropriate termination condition by your own.
53  	 */
54  	@Deprecated
55  	@Override
56  	public TerminationStrategy getTerminationStrategy() {
57  		return TerminationStrategy.BY_SELF_DECISION;
58  	}
