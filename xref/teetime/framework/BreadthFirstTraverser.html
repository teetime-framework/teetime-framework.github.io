3   import java.util.HashSet;
4   import java.util.LinkedList;
5   import java.util.Queue;
6   import java.util.Set;
7   
8   public class BreadthFirstTraverser {
9   
10  	public void traverse(final AbstractStage startStage, final ITraverserVisitor traverserVisitor) {
11  		final Set<AbstractStage> visitedStages = new HashSet<AbstractStage>();
12  		final Queue<AbstractStage> queue = new LinkedList<>();
13  
14  		queue.add(startStage);
15  		visitedStages.add(startStage);
16  
17  		while (!queue.isEmpty()) {
18  			AbstractStage stage = queue.remove();
19  			traverserVisitor.visit(startStage);
20  
21  			for (OutputPort<?> outputPort : stage.getOutputPorts()) {
22  				AbstractStage targetStage = outputPort.getPipe().getTargetPort().getOwningStage();
23  
24  				if (!visitedStages.contains(targetStage)) {
25  					queue.add(targetStage);
26  					visitedStages.add(startStage);
27  				}
28  
29  				traverserVisitor.visit(outputPort);
30  			}
31  		}
32  	}
33  }
