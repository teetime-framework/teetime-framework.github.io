16  package teetime.framework.test;
17  
18  import java.util.Arrays;
19  import java.util.HashMap;
20  import java.util.List;
21  import java.util.Map;
22  
23  import org.hamcrest.Factory;
24  import org.hamcrest.Matcher;
25  
26  import teetime.framework.*;
27  
28  /**
29   * This class can be used to test single stages in JUnit test cases.
30   *
31   * @author Christian Wulf, Nils Christian Ehmke
32   */
33  public class StageTester {
34  
35  	private final StageUnderTest stageUnderTest;
36  	private final Map<InputPort<Object>, List<Object>> inputElementsByPort = new HashMap<>();
37  	private final Map<OutputPort<Object>, List<Object>> outputElementsByPort = new HashMap<>();
38  
39  	private StageTester(final StageUnderTest stageUnderTest) {
40  		this.stageUnderTest = stageUnderTest;
41  	}
42  
43  	/**
44  	 * Prepares to test the given stage.
45  	 *
46  	 * @param stage
47  	 *            to be tested
48  	 * @return
49  	 * 		a stage test builder
50  	 */
51  	public static StageTester test(final AbstractStage stage) {
52  		StageUnderTest stageUnderTest = new PrimitiveStageUnderTest(stage);
53  
54  		return new StageTester(stageUnderTest);
55  	}
56  
57  	/**
58  	 * Prepares to test the given composite stage.
59  	 *
60  	 * @param compositeStage
61  	 *            to be tested
62  	 * @return
63  	 * 		a stage test builder
64  	 */
65  	public static StageTester test(final CompositeStage compositeStage) {
66  		StageUnderTest stageUnderTest = new CompositeStageUnderTest(compositeStage);
67  
68  		return new StageTester(stageUnderTest);
69  	}
70  
71  	/**
72  	 * @param elements
73  	 *            which serve as input. If nothing should be sent, pass
74  	 */
75  	@SafeVarargs
76  	public final <I> InputHolder<I> send(final I... elements) {
77  		return this.send(Arrays.asList(elements));
78  	}
79  
80  	/**
81  	 * @param elements
82  	 *            which serve as input. If nothing should be sent, pass
83  	 *
84  	 *            <pre>
85  	 * Collections.&lt;your type&gt;emptyList().
86  	 *            </pre>
87  	 */
88  	public <I> InputHolder<I> send(final List<I> elements) {
89  		final InputHolder<I> inputHolder = new InputHolder<I>(this, elements);
90  		return inputHolder;
91  	}
92  
93  	/**
94  	 * @param actualElements
95  	 *            which should be tested against the expected elements.
96  	 *
97  	 * @deprecated since 3.0. Use the following code instead:
98  	 *
99  	 *             <pre>
100 	 * {@code
101 	 * import static StageTester.*;
102 	 * ...
103 	 * assertThat(stage.getOutputPort(), produces(1,2,3));
104 	 * }
105 	 *             </pre>
106 	 *
107 	 */
108 	@Deprecated
109 	public <O> OutputHolder<O> receive(final List<O> actualElements) {
110 		final OutputHolder<O> outputHolder = new OutputHolder<O>(this, actualElements);
111 		return outputHolder;
112 	}
113 
114 	/**
115 	 * Does nothing. Can be used to make the test more readable.
116 	 */
117 	public StageTester and() {
118 		return this;
119 	}
120 
121 	/**
122 	 * This method will start the test and block until it is finished.
123 	 *
124 	 * @throws ExecutionException
125 	 *             if at least one exception in one thread has occurred within the analysis.
126 	 *             The exception contains the pairs of thread and throwable.
127 	 *
128 	 */
129 	public void start() {
130 		final Configuration configuration = new TestConfiguration(this);
131 		final Execution<Configuration> analysis = new Execution<Configuration>(configuration);
132 		analysis.executeBlocking();
133 	}
134 
135 	/* default */ StageUnderTest getStageUnderTest() {
136 		return stageUnderTest;
137 	}
138 
139 	/* default */ Map<InputPort<Object>, List<Object>> getInputElementsByPort() {
140 		return inputElementsByPort;
141 	}
142 
143 	/* default */ Map<OutputPort<Object>, List<Object>> getOutputElementsByPort() {
144 		return outputElementsByPort;
145 	}
146 
147 	@SafeVarargs
148 	@Factory
149 	public static <T> Matcher<OutputPort<T>> produces(final T... values) {
150 		return new Produces<T, OutputPort<T>>(values);
151 	}
152 
153 	@Factory
154 	public static <T> Matcher<OutputPort<T>> producesNothing() {
155 		return new Produces<T, OutputPort<T>>();
156 	}
157 
158 }
