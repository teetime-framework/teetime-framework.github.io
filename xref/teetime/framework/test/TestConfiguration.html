3   import java.util.ArrayList;
4   import java.util.List;
5   import java.util.Map;
6   
7   import teetime.framework.Configuration;
8   import teetime.framework.InputPort;
9   import teetime.framework.OutputPort;
10  import teetime.framework.pipe.DummyPipe;
11  import teetime.stage.CollectorSink;
12  import teetime.stage.InitialElementProducer;
13  
14  class TestConfiguration extends Configuration {
15  
16  	@SuppressWarnings({ "PMD.AvoidInstantiatingObjectsInLoops", "unchecked" })
17  	public TestConfiguration(final StageTester stageTester) {
18  		StageUnderTest stageUnderTest = stageTester.getStageUnderTest();
19  
20  		Map<InputPort<Object>, List<Object>> inputElementsByPort = stageTester.getInputElementsByPort();
21  		for (InputPort<?> inputPort : stageUnderTest.getInputPorts()) {
22  			if (inputPort.getPipe() != null) {
23  				continue; // skip reflexive/loop pipe
24  			}
25  			List<Object> inputElements = inputElementsByPort.getOrDefault(inputPort, new ArrayList<>());
26  			final InitialElementProducer<?> producer = StageFactory.createProducer(inputElements);
27  			connectPorts(producer.getOutputPort(), (InputPort<Object>) inputPort);
28  		}
29  
30  		// declareActive that works for both AbstractStage and CompositeStage
31  		for (InputPort<?> inputPort : stageUnderTest.getInputPorts()) {
32  			inputPort.getOwningStage().declareActive();
33  		}
34  
35  		Map<OutputPort<Object>, List<Object>> outputElementsByPort = stageTester.getOutputElementsByPort();
36  		for (OutputPort<?> outputPort : stageUnderTest.getOutputPorts()) {
37  			if (outputPort.getPipe() != DummyPipe.INSTANCE) {
38  				continue; // skip reflexive/loop pipe
39  			}
40  			List<Object> outputElements = outputElementsByPort.getOrDefault(outputPort, new ArrayList<>());
41  			final CollectorSink<Object> sink = StageFactory.createSink(outputElements);
42  			connectPorts(outputPort, sink.getInputPort());
43  		}
44  	}
45  
46  }
