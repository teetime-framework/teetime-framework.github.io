16  package teetime.framework;
17  
18  import teetime.framework.signal.ISignal;
19  
20  /**
21   * Represents an unsynchronized pipe which can be used to connect stages within the same thread.
22   *
23   * @author Christian Wulf (chw)
24   *
25   * @param <T>
26   *            the type of the elements which this pipe should transfer.
27   */
28  public abstract class AbstractUnsynchedPipe<T> extends AbstractPipe<T> {
29  
30  	private boolean closed;
31  
32  	protected AbstractUnsynchedPipe(final OutputPort<? extends T> sourcePort, final InputPort<T> targetPort, final int capacity) {
33  		super(sourcePort, targetPort, capacity);
34  	}
35  
36  	@Override
37  	public boolean addNonBlocking(final Object element) {
38  		return add(element);
39  	}
40  
41  	@Override
42  	public final void sendSignal(final ISignal signal) {
43  		// getTargetPort is always non-null since the framework adds dummy ports if necessary
44  		this.cachedTargetStage.onSignal(signal, this.getTargetPort());
45  	}
46  
47  	@Override
48  	public final void reportNewElement() {
49  		this.cachedTargetStage.executeStage();
50  	}
51  
52  	@Override
53  	public boolean isClosed() {
54  		return closed;
55  	}
56  
57  	@Override
58  	public void close() {
59  		closed = true;
60  	}
61  
62  	@SuppressWarnings("PMD.EmptyMethodInAbstractClassShouldBeAbstract")
63  	@Override
64  	public void waitForStartSignal() throws InterruptedException {
65  		// do nothing
66  	}
67  
68  }
